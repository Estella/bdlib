  23 #ifndef _BD_BASE64_H
  24 #define _BD_BASE64_H 1

  26 #include "bdlib.h"

  28 BDLIB_NS_BEGIN
  29 class String;
  30 
  31 class Base64 {
  32 };
  33 
  34 /**
  35   * @brief Encode a plaintext string into base64 (returns a buffer)
  36   * @param src A c-style string to encode
  37   * @param len Reference to length of string (to be updated on return)
  38   * @return An encoded NULL-terminated c-style string (must be free()d later)
  39   */
  40 char *b64enc(const unsigned char *src, size_t *len);
  41 
  42 /**
  43  * @brief Base64 encode a string
  44  * @param string The string to encode
  45  * @return A new, encoded string
  46  */
  47 String base64Encode(const String&);
  48 
  49 /**
  50   * @brief Encode a plaintext string into base64 (using a given buffer)
  51   * @param data The c-style string to encode
  52   * @param len Reference to length of string (to be updated on return)
  53   * @param dest Reference to the buffer to encode into
  54   */
  55 void b64enc_buf(const unsigned char *data, size_t *len, char *dest);
  56 
  57 /**
  58   * @brief Decode a base64 encoded string into plaintext (returns a buffer)
  59   * @param src A c-style string to decode
  60   * @param len Reference to length of string (to be updated on return)
  61   * @return A decoded NULL-terminated c-style string (must be free()d later)
  62   */
  63 char *b64dec(const unsigned char *data, size_t *len);
  64 
  65 /**
  66  * @brief Base64 decode a string
  67  * @param string The string to decode
  68  * @return A new, decoded string
  69  */
  70 String base64Decode(const String&);
  71 
  72 /**
  73   * @brief Decode a base64 encoded string into plaintext (using a given buffer)
  74   * @param data The c-style string to decode
  75   * @param len Reference to length of string (to be updated on return)
  76   * @param dest Reference to the buffer to decode into
  77   */
  78 void b64dec_buf(const unsigned char *data, size_t *len, char *dest);
  79 
  80 BDLIB_NS_END
  81 #endif /* !_BD_BASE64_H */
