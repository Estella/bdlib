  32 template <class Key, class Value>
  33 /**
  34   * @class HashTable
  35   * @brief HashTable data structure
  36   * @todo replace()
  37   * @todo iterators
  38   * @todo resizing/growing
  39   *
  40   */
  41 class HashTable {
  42   private:
  43     typedef KeyValue<Key, Value> iterator_type;
  44     typedef hash<Key> hasher;
  45 
  46     List<iterator_type> *list;
  47     size_t _size;
  48     size_t _capacity;
  49     hasher _hash;
  50 
  51     inline size_t getIndex(const Key &key) const {
  52       return _hash(key) % _capacity;
  53     }
  54   public:
  55     HashTable() : list(new List<iterator_type>[100]), _size(0), _capacity(100), _hash() {};
  56     explicit HashTable(size_t capacity_in) : list(new List<iterator_type>[capacity_in]), _size(0), _capacity(capacity_in), _hash() {};
  57     HashTable(const HashTable<Key, Value> &table) : list(new List<iterator_type>[table._capacity]), _size(table._size), _capacity(table._capacity), _hash(table._hash) {
  58       for (size_t i = 0; i < _capacity; ++i)
  59           list[i] = table.list[i];
  60     };
  61 
  62     virtual ~HashTable() {
  63       delete[] list;
  64     }
  65 
  66     void clear() {
  67       for (size_t i = 0; i < _capacity; ++i)
  68         list[i].clear();
  69       _size = 0;
  70     }
  71 
  72     HashTable &operator = (const HashTable<Key, Value> &table) { 
  73       if (&table != this) {
  74         delete[] list;
  75         list = new List<iterator_type>[_capacity];
  76         for (size_t i = 0; i < _capacity; ++i) 
  77           list[i] = table.list[i];
  78         _size = table._size;
  79         _capacity = table._capacity;
  80         _hash = table._hash;
  81       }
  82       return *this; 
  83     }
  84 
  85     inline size_t size() const { return _size; };
  86     inline size_t capacity() const { return _capacity; };
  87     inline bool isEmpty() const { return size() == 0; };
  88     //inline operator bool() const { return !isEmpty(); };
  89 
  90     bool insert(const Key &key, const Value &value) { 
  91       if (contains(key)) return false;
  92       list[getIndex(key)] << iterator_type(key, value);
  93       ++_size;
  94       return true;
  95     }
  96 
  97     inline bool contains(const Key &key) const {
  98       if (isEmpty()) return false;
  99       return list[getIndex(key)].contains(iterator_type(key, Value()));
 100     };
 101 
 102     bool remove(const Key &key) {
 103       if (isEmpty()) return false;
 104       if (list[getIndex(key)].remove(iterator_type(key, Value()))) {
 105         --_size;
 106         return true;
 107       }
 108       return false;
 109     };
 110 
 111     inline Value getValue(const Key &key) const {
 112       Value empty;
 113       if (isEmpty()) return empty;
 114       return list[getIndex(key)].find(iterator_type(key, Value())).value();
 115     };
 116 
 117     /**
 118       * @brief Associate array type accessor (rvalue)
 119       * @param key The key to search for
 120       */
 121     inline const Value operator [](const Key& key) const { return getValue(key); }
 122     
 123     /**
 124       * @brief Associate array type accessor (lvalue)
 125       * @param key The key to search for
 126       * @sa find_or_insert_key 
 127       * @code table["key"] = "value";
 128       * If the key is not in the table, it is inserted, and the value set to the rvalue given.
 129       */
 130     inline Value& operator [](const Key& key) {
 131       return find_or_insert_key(key);
 132     }
 133  
 134     /**
 135      * @brief Find a key in the list or insert it.
 136      */
 137     inline Value& find_or_insert_key(const Key& key) {
 138       if (!contains(key))
 139         insert(key, Value());
 140       return list[getIndex(key)].findRef(iterator_type(key, Value())).v;
 141     }
 142 };
