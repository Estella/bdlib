  33 template <class Key, class Value>
  34 /**
  35   * @class HashTable
  36   * @brief HashTable data structure
  37   * @todo replace()
  38   * @todo iterators
  39   * @todo resizing/growing
  40   *
  41   */
  42 class HashTable {
  43   private:
  44     typedef KeyValue<Key, Value> iterator_type;
  45     typedef hash<Key> hasher;
  46 
  47     List<iterator_type> *list;
  48     size_t _size;
  49     size_t _capacity;
  50     hasher _hash;
  51 
  52     inline size_t getIndex(const Key &key) const {
  53       return _hash(key) % _capacity;
  54     }
  55   public:
  56     HashTable() : list(new List<iterator_type>[100]), _size(0), _capacity(100), _hash() {};
  57     explicit HashTable(size_t capacity) : list(new List<iterator_type>[capacity]), _size(0), _capacity(capacity), _hash() {};
  58     HashTable(const HashTable<Key, Value> &table) : list(new List<iterator_type>[table._capacity]), _size(table._size), _capacity(table._capacity), _hash(table._hash) {
  59       for (size_t i = 0; i < _capacity; ++i)
  60           list[i] = table.list[i];
  61     };
  62 
  63     virtual ~HashTable() {
  64       delete[] list;
  65     }
  66 
  67     void clear() {
  68       for (size_t i = 0; i < _capacity; ++i)
  69         list[i].clear();
  70       _size = 0;
  71     }
  72 
  73     HashTable &operator = (const HashTable<Key, Value> &table) { 
  74       if (&table != this) {
  75         delete[] list;
  76         list = new List<iterator_type>[_capacity];
  77         for (size_t i = 0; i < _capacity; ++i) 
  78           list[i] = table.list[i];
  79         _size = table._size;
  80         _capacity = table._capacity;
  81         _hash = table._hash;
  82       }
  83       return *this; 
  84     }
  85 
  86     inline size_t size() const { return _size; };
  87     inline size_t capacity() const { return _capacity; };
  88     inline bool isEmpty() const { return size() == 0; };
  89     //inline operator bool() const { return !isEmpty(); };
  90 
  91     bool insert(const Key &key, const Value &value) { 
  92       if (contains(key)) return false;
  93       list[getIndex(key)] << iterator_type(key, value);
  94       ++_size;
  95       return true;
  96     }
  97 
  98     inline bool contains(const Key &key) const {
  99       if (isEmpty()) return false;
 100       return list[getIndex(key)].contains(iterator_type(key, Value()));
 101     };
 102 
 103     bool remove(const Key &key) {
 104       if (isEmpty()) return false;
 105       if (list[getIndex(key)].remove(iterator_type(key, Value()))) {
 106         --_size;
 107         return true;
 108       }
 109       return false;
 110     };
 111 
 112     inline Value getValue(const Key &key) const {
 113       Value empty;
 114       if (isEmpty()) return empty;
 115       return list[getIndex(key)].find(iterator_type(key, Value())).value();
 116     };
 117 
 118     /**
 119       * @brief Associate array type accessor (rvalue)
 120       * @param key The key to search for
 121       */
 122     inline const Value operator [](const Key& key) const { return getValue(key); }
 123     
 124     /**
 125       * @brief Associate array type accessor (lvalue)
 126       * @param key The key to search for
 127       * @sa find_or_insert_key 
 128       * @code table["key"] = "value";
 129       * If the key is not in the table, it is inserted, and the value set to the rvalue given.
 130       */
 131     inline Value& operator [](const Key& key) {
 132       return find_or_insert_key(key);
 133     }
 134  
 135     /**
 136      * @brief Find a key in the list or insert it.
 137      */
 138     inline Value& find_or_insert_key(const Key& key) {
 139       if (!contains(key))
 140         insert(key, Value());
 141       return list[getIndex(key)].findRef(iterator_type(key, Value())).v;
 142     }
 143 };
