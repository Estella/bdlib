  30 template <class T>
  31 /**
  32   * @class List
  33   * @brief List data structure
  34   */
  35 class List {
  36   class ListIterator;
  37 
  38   public:
  39     typedef ListIterator iterator;
  40     typedef T iterator_type;
  41 
  42 
  43   protected:
  44     struct Node {
  45       iterator_type ptr;
  46       Node* next;
  47       Node* prev;
  48       
  49       Node(const iterator_type& p) : ptr(p), next(NULL), prev(NULL) {};
  50     };
  51 
  52     void deleteNode(Node* node) {
  53       Node* next = node->next; //NULL if node = tail
  54       Node* prev = node->prev; //NULL if node = head
  55 
  56       delete node;
  57 
  58       if (next) next->prev = prev;
  59       if (prev) prev->next = next;
  60 
  61       if (node == head) head = next;
  62       if (node == tail) tail = prev;
  63 
  64       --my_size;
  65     }
  66 
  67     Node* head;
  68     Node* tail;
  69     size_t my_size;
  70 
  71   public:
  72     List() : head(NULL), tail(NULL), my_size(0) {};
  73     virtual ~List() { clear(); };
  74 
  75     List(const List& list) : head(NULL), tail(NULL), my_size(0) {
  76       for (Node* search = list.head; search; search = search->next)
  77         insert(search->ptr);
  78     }
  79 
  80     List& operator =(const List& list) {
  81       if (&list != this) {
  82         clear();
  83         for (Node* search = list.head; search; search = search->next)
  84           insert(search->ptr);
  85       }
  86       return *this;
  87     }
  88 
  89     void clear() {
  90       Node* node = head, *node_n = NULL;
  91       while (node) {
  92         node_n = node->next;
  93         delete node;
  94         node = node_n;
  95       }
  96       head = tail = NULL;
  97       my_size = 0;
  98     };
  99 
 100     inline size_t size() const { return my_size; };
 101     inline bool isEmpty() const { return size() == 0; };
 102     //inline operator bool() const { return !isEmpty(); };
 103 
 104     /**
 105      * @brief Insert into the list at the head
 106      * @param ptr The ptr to insert
 107      */
 108     void insert(const iterator_type& ptr) {
 109       Node* node = new Node(ptr);
 110       if (!head) {
 111         head = tail = node;
 112       } else {
 113         head->prev = node;
 114         node->next = head;
 115         head = node;
 116       }
 117       ++my_size;
 118     }
 119 
 120     /**
 121      * @sa insert
 122      */
 123     inline friend List<iterator_type>& operator<< (List<iterator_type>& list, const iterator_type& item) {
 124       list.insert(item);
 125       return list;
 126     }
 127 
 128     bool contains(const iterator_type& ptr) const {
 129       if (head) {
 130         for (Node* node = head; node; node = node->next) {
 131           if (node->ptr == ptr)
 132             return 1;
 133         }
 134       }
 135       return 0;
 136     }
 137 
 138    const iterator_type find(const iterator_type& ptr) const {
 139       const iterator_type empty;
 140       if (isEmpty()) return empty;
 141 
 142       for (Node* current = head; current; current = current->next) {
 143         if (current->ptr == ptr)
 144           return current->ptr;
 145       }
 146       return empty;
 147     }
 148 
 149     /**
 150      * @brief Return a reference to the found element
 151      */
 152     iterator_type& findRef(const iterator_type& ptr) {
 153       Node* current = head;
 154       for (; current; current = current->next) {
 155         if (current->ptr == ptr)
 156           break;
 157       }
 158       return current->ptr;
 159     }
 160 
 161     bool remove(const iterator_type& ptr) {
 162       if (head) {
 163       /* cases (removing X):
 164        * 1) list = (X)
 165        * 2) list = (X)->(Y)
 166        * 3) list = (W)->(X)
 167        * 4) list = (W)->(X)->(Y)
 168        */
 169         for (Node* node = head; node; node = node->next) {
 170           if (node->ptr == ptr) {
 171             deleteNode(node);
 172             return true;
 173           }
 174         }
 175       }
 176       return false;
 177     }
 178 
 179     private:
 180     class ListIterator : public Iterator<iterator_type> {
 181       friend class List;
 182       private:
 183         List *list;
 184         Node *current;
 185         bool end;
 186 
 187         ListIterator(List &_list, bool _end = 0) : Iterator<iterator_type>(),
 188                                                               list(&_list),
 189                                                               current(_end ? _list.head : _list.tail), /* tail is beginning */
 190                                                               end(_end) {
 191         };
 192       public:
 193         ListIterator(const iterator& iter) : Iterator<iterator_type>(),
 194                                              list(iter.list),
 195                                              current(iter.current),
 196                                              end(iter.end) {
 197         };
 198         ListIterator() : Iterator<iterator_type>(),
 199                          list(NULL),
 200                          current(NULL),
 201                          end(0) {
 202         };
 203 
 204         iterator& operator =(const iterator& iter) {
 205           list = iter.list;
 206           current = iter.current;
 207           end = iter.end;
 208           return *this;
 209         }
 210 
 211         virtual ~ListIterator() { }
 212 
 213         virtual void remove() {
 214           Node *nextValue = end == 0 ? current->next : current->prev;
 215 
 216           list->deleteNode(current);
 217 
 218           current = nextValue;
 219         }
 220 
 221         virtual operator bool() { return (current != NULL); };
 222         virtual operator iterator_type () { return operator*(); };
 223           
 224         virtual iterator_type& operator *() { return (iterator_type&)current->ptr; }
 225 
 226         //Postfix
 227         virtual iterator operator ++(int) {
 228           iterator tmp(*this);
 229           current = current->prev;
 230           return tmp;
 231         }
 232 
 233         virtual iterator& operator ++() {
 234           current = current->prev;
 235           return *this;
 236         }
 237 
 238         //Postfix
 239         virtual iterator operator --(int) {
 240           iterator tmp(*this);
 241           current = current->next;
 242           return tmp;
 243         }
 244 
 245         virtual iterator& operator --() {
 246           current = current->next;
 247           return *this;
 248         }
 249     };
 250   public:
 251     inline iterator begin() { return iterator(*this); };
 252     inline iterator end() { return iterator(*this, 1); };
 253 };
