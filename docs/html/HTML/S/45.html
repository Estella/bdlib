  55 void b64enc_buf(const unsigned char *data, size_t *len, char *dest)
  56 {
  57   char *buf = dest;
  58 
  59   /* Encode 3 bytes at a time. */
  60 
  61   /*
  62    *
  63    * |       0       |       1       |       2       |
  64    *
  65    * |               |               |               |
  66    * |       |       |       |       |       |       |
  67    * |   |   |   |   |   |   |   |   |   |   |   |   |
  68    * | | | | | | | | | | | | | | | | | | | | | | | | |
  69    *  7 6 5 4 3 2 1 0 7 6 5 4 3 2 1 0 7 6 5 4 3 2 1 0
  70    * -------------------------------------------------
  71    *  5 4 3 2 1 0 5 4 3 2 1 0 5 4 3 2 1 0 5 4 3 2 1 0 
  72    * | | | | | | | | | | | | | | | | | | | | | | | | |
  73    * |   | | |   |   | | |   |   | | |   |   | | |   |
  74    * |     |     |     |     |     |     |     |     |
  75    * |           |           |           |           |
  76    *
  77    * |     0     |     1     |     2     |     3     |
  78    *
  79    */
  80 
  81 
  82   while (*len >= NUM_ASCII_BYTES) {
  83     /* buf[0] is the 6 left-most bits of data[0] */
  84     buf[0] = b64_charset[(data[0] & 0xfc) >> 2];
  85     /* buf[1] is the right-most 2 bits of data[0] and the left-most 4 bits of data[1] */
  86     buf[1] = b64_charset[((data[0] & 0x03) << 4) | ((data[1] & 0xf0) >> 4)];
  87     /* buf[2] is the right-most 4 bits of data[1] and the 2 left-most bits of data[2] */
  88     buf[2] = b64_charset[((data[1] & 0x0f) << 2) | ((data[2] & 0xc0) >> 6)];
  89     /* buf[3] is the right-most 6 bits of data[2] */
  90     buf[3] = b64_charset[data[2] & 0x3f];
  91 
  92     data += NUM_ASCII_BYTES;
  93     buf += NUM_ENCODED_BYTES;
  94     *len -= NUM_ASCII_BYTES;
  95   }
  96 
  97   /* There is either 1 or 2 bytes left to encode (and possibly some padding to add) */
  98   if (*len > 0) {
  99     buf[0] = b64_charset[(data[0] & 0xfc) >> 2];
 100     if (*len == 1) {
 101       buf[1] = b64_charset[((data[0] & 0x03) << 4)];
 102       buf[2] = PADDING_CHAR;
 103     } else if (*len > 1) { /* == 2 .. left as >1 to help optimize with last if */
 104       /* Tack on to buf[1] the left-most 4 bits of data[1] */
 105       buf[1] = b64_charset[((data[0] & 0x03) << 4) | ((data[1] & 0xf0) >> 4)];
 106       buf[2] = b64_charset[(data[1] & 0x0f) << 2];
 107     }
 108     buf[3] = PADDING_CHAR;
 109     buf += NUM_ENCODED_BYTES;
 110   }
 111 
 112   *len = buf - dest;
 113 }

 123 char *b64enc(const unsigned char *src, size_t *len)
 124 {
 125   /* Take the length and round up the next 4-byte boundary */
 126   size_t dlen = (((*len + (NUM_ASCII_BYTES - 1)) / NUM_ASCII_BYTES) * NUM_ENCODED_BYTES);
 127   char *dest = (char *) malloc(dlen + 1);;
 128 
 129   b64enc_buf(src, len, dest);
 130   dest[*len] = '\0';
 131   return dest;
 132 }

 134 void b64dec_buf(const unsigned char *data, size_t *len, char *dest)
 135 {
 136   char *buf = dest;
 137   size_t blocks = (*len / NUM_ENCODED_BYTES) + ((*len % NUM_ENCODED_BYTES) ? 1 : 0);
 138 
 139   /* Convert the encoded base64 character back into our base255 ASCII character */
 140   while (blocks > 0) {
 141     /* buf[0] is the 6 bits from data[0] + left-most 2 bits of data[1] */
 142     *buf++ = (b64_indexes[data[0]] << 2) + ((b64_indexes[data[1]] & 0x30) >> 4);  /* 0x30 not required but to be clear. */
 143     if (data[2] != PADDING_CHAR) {
 144       /* buf[1] is the right-most 4 bits of data[1] + left-most 4 bits of data[2] */
 145       *buf++ = ((b64_indexes[data[1]] & 0x0f) << 4) + ((b64_indexes[data[2]] & 0x3c) >> 2);
 146       if (data[3] != PADDING_CHAR) {
 147         /* buf[2] is the right-most 2 bits of data[2] + data[3] */
 148         *buf++ = ((b64_indexes[data[2]] & 0x03) << 6) + b64_indexes[data[3]];
 149       }
 150     }
 151     data += NUM_ENCODED_BYTES;
 152     --blocks;
 153   }
 154 
 155   *len = (buf - dest);
 156 }

 158 char *b64dec(const unsigned char *data, size_t *len)
 159 {
 160   size_t dlen = ((*len / NUM_ENCODED_BYTES) + ((*len % NUM_ENCODED_BYTES) ? 1 : 0)) * NUM_ASCII_BYTES;
 161   char *dest = (char *) malloc(dlen + 1);
 162 
 163   b64dec_buf(data, len, dest);
 164   dest[*len] = '\0';
 165   return dest;
 166 }
