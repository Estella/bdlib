  40 /**
  41  * @brief Ensure that the buffer capacity() is >= newSize; else grow/copy into larger buffer.
  42  * @param newSize A size that we need to Allocate the buffer to.
  43  * @pre newSize is > 0 (assumed as size_t is unsigned)
  44  * @post The buffer is at least nsize bytes long.
  45  * @post If the buffer had to grow, the old data was deep copied into the new buffer and the old deleted.
  46  */
  47 void StringBuf::Reserve(const size_t newSize) const
  48 {
  49   /* Don't new if we already have enough room! */
  50   if (size < newSize) { 
  51     size = std::max(size_t(size * 1.5), newSize);
  52 
  53     char *newbuf = AllocBuf(size);
  54 
  55     if (newbuf != buf) {
  56       std::copy(buf, buf + len, newbuf);
  57       FreeBuf(buf);
  58       buf = newbuf;
  59     }
  60   }
  61 }

  63 /**
  64  * @brief Ensure that our internal buffer is unshared.
  65  * @param n Create/Grow the buffer to this size.
  66  * @pre n is > 0; this is assumed due to size_t being unsigned though.
  67  * @post The internal buffer/data is unshared
  68  * @post The buffer is at least size n.
  69  * @post The buffer is deep copied to a new buffer.
  70  * 
  71  * Ensure that our internal buffer is unshared.
  72  * If needed, performs a deep copy into a new buffer (COW).
  73  * Also take a hint size n of the new string's size as to avoid needless copying/allocing.
  74  */
  75 void String::COW(size_t n) const {
  76   const char *p = constBuf();
  77   size_t oldLength = length();
  78   size_t oldCapacity = capacity();
  79 
  80   doDetach();
  81   Reserve( std::max(oldCapacity, n) ); //Will set capacity()/size
  82   std::copy(p, p + oldLength, Buf());
  83   setLength(oldLength);
  84 }

  87 /**
  88  * @brief Compare our String object with another String object, but only n characters
  89  * @param str The String object to compare to.
  90  * @param n The number of characters to compare.
  91  * @return an integer less than, equal to, or greater than zero if our buffer is found, respectively, to be less than, to match, or be greater than str.
  92 */
  93 int String::compare(const String &str, size_t n) const
  94 {
  95   size_t my_len = length();
  96   size_t slen = std::min(str.length(), n);
  97   size_t len = std::min(my_len, slen);
  98 
  99 
 100   for (size_t i = 0; i < len; ++i) {
 101     if ((*this)[i] < str[i])
 102       return -1;
 103     else if (str[i] < (*this)[i])
 104       return 1;
 105   }
 106   return my_len - slen;
 107 }

 161   std::copy(string, string + slen, Ref->buf + k);
 162   addLength(slen);

 189 /**
 190  * @brief Inserts a String object into our buffer
 191  * @param k The index to insert at.
 192  * @param string The string to insert.
 193  * @param n The length to insert.
 194  * @post The buffer contains n characters from string inserted at index k.
 195  */
 196 void String::insert(int k, const String &string, int n) {
 197   if (n == 0) return;
 198   if (k && !hasIndex(k-1)) return;
 199   
 200   int slen = string.length();
 201 
 202   if (n == -1 || n > slen)
 203     n = slen;
 204   slen -= slen - n;
 205   AboutToModify(length() + slen);
 206   memmove(Ref->buf + k + slen, Ref->buf + k, length() - k);
 207   std::copy(string.begin(), string.begin() + slen, Ref->buf + k);
 208   addLength(slen);
 209 }

 283 std::istream& operator >> (std::istream& is, String &string) {
 284   char ch;
 285   string = "";    // empty string, will build one char at-a-time
 286   is >> ch;    // whitespace skipped, first non-white char in ch
 287 
 288   if (!is.fail())
 289   {
 290     do
 291     {
 292       string += ch;
 293       is.get(ch);
 294     } while (!is.fail() && !isspace(ch));
 295 
 296     if (isspace(ch))    // put whitespace back on the stream
 297     {
 298       is.putback(ch);
 299     }
 300   }
 301   return is;
 302 }

 304 std::istream& getline(std::istream& is, String &string) {   
 305  
 306   char ch;
 307   string = "";     // empty string, will build one char at-a-time
 308  
 309   while (is.get(ch) && ch != '\n')
 310     string += ch;
 311   
 312   return is;
 313 }

 315 /**
 316  * @brief Return a new string from a substring
 317  * @return a new String
 318  * @param start The offset to begin the substring from (indexed from 0)
 319  * @param len The length of the substring to return
 320  * The returned substring is a reference to the original string until modified.
 321  */
 322 String String::substring(int start, int len) const
 323 {
 324   // Start is after the end, return an empty string
 325   if (start >= (signed) length())
 326     return String("");
 327  
 328   String newString(*this);
 329 
 330   //Count backwards from the end
 331   if (start < 0)
 332     start = length() + start;
 333   // Start was before the beginning, just reset to the beginning
 334   if (start < 0)
 335     start = 0;
 336 
 337   newString.offset = start;
 338   //If the length of the substring exceeds the end of the string, truncate to the end of the string
 339   if (start + len >= (signed) length())
 340     len = length() - start;
 341   // If the length is negative, stop from counting backwards from the end
 342   else if (len < 0)
 343     len = length() - start + len;
 344   newString.setLength(len);
 345   return newString;
 346 }

 394 /**
 395  * @brief DJB's hash function
 396  */
 397 unsigned int String::hash() const {
 398   unsigned int hash = 5381;
 399 
 400   for(size_t i = 0; i < length(); ++i)
 401     hash = ((hash << 5) + hash) + data()[i];
 402   return (hash & 0x7FFFFFFF);
 403 }
