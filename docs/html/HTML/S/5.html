  24 #ifndef _BD_THREAD_H
  25 #define _BD_THREAD_H 1
  26 
  27 #include "bdlib.h"
  28 
  29 #if !defined(USE_PTHREAD) && !defined(WIN32)
  30 # define NO_THREADS 1
  31 #endif
  32 
  33 #if defined(USE_PTHREAD)
  34 # include <pthread.h>
  35 typedef void* threadMainRet_t;
  36 typedef pthread_t threadHandle_t;
  37 #elif defined(WIN32)
  38 # include <windows.h>
  39 typedef unsigned int __stdcall threadMainRet_t;
  40 typedef HANDLE threadHandle_t;
  41 #elif defined(NO_THREADS) /* NO THREAD SUPPORT */
  42 typedef void* threadMainRet_t;
  43 typedef short threadHandle_t;
  44 #endif
  45 
  46 typedef unsigned int threadId_t;
  47 
  48 BDLIB_NS_BEGIN
  49 
  50 #define THREAD_STARTED  0x1
  51 #define THREAD_DETACHED 0x2
  52 
  53 static threadMainRet_t threadMain(void*);
  54 
  55 /**
  56   * @class Thread
  57   * @brief Extendable thread class
  58   */
  59 class Thread {
  60   private:
  61     threadHandle_t handle;
  62     threadId_t id;
  63     short status;
  64     void* param;
  65 
  66   friend threadMainRet_t threadMain(void*);
  67   public:
  68     Thread(void) : handle(0), id(0), status(0), param(NULL) {};
  69     virtual ~Thread() { stop(); };
  70     int start(void* = NULL);
  71     void detach(void);
  72     void* wait(void);
  73     void stop(void);
  74     threadId_t getThreadId(void);
  75     static threadId_t getCurrentThreadId(void);
  76     static void sleep(int);
  77   protected: 
  78     virtual void* run(void*)=0;
  79   private:
  80     inline bool isStarted() { return status & THREAD_STARTED; };
  81     inline void setStarted() { status |= THREAD_STARTED; };
  82     inline bool isDetached() { return status & THREAD_DETACHED; };
  83     inline void setDetached() { status |= THREAD_DETACHED; };
  84     /**
  85      * @brief Is the thread started and not detached?
  86      * If status == THREAD_STARTED (1), then the THREAD_DETACHED bit is not set, 
  87      * so this is equivalent to: isStarted() && !isDetached()
  88      */
  89     inline bool shouldDetach() { return status == THREAD_STARTED; };
  90 };
  91 
  92 BDLIB_NS_END

  94 #endif /* !_BD_THREAD_H */
