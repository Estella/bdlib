  27 #include "bdlib.h"

  29 BDLIB_NS_BEGIN
  30 class String;
  31 
  32 class Base64 {
  33 };
  34 
  35 /**
  36   * @brief Encode a plaintext string into base64 (returns a buffer)
  37   * @param src A c-style string to encode
  38   * @param len Reference to length of string (to be updated on return)
  39   * @return An encoded NULL-terminated c-style string (must be free()d later)
  40   */
  41 char *b64enc(const unsigned char *src, size_t *len);
  42 
  43 /**
  44  * @brief Base64 encode a string
  45  * @param string The string to encode
  46  * @return A new, encoded string
  47  */
  48 String base64Encode(const String&);
  49 
  50 /**
  51   * @brief Encode a plaintext string into base64 (using a given buffer)
  52   * @param data The c-style string to encode
  53   * @param len Reference to length of string (to be updated on return)
  54   * @param dest Reference to the buffer to encode into
  55   */
  56 void b64enc_buf(const unsigned char *data, size_t *len, char *dest);
  57 
  58 /**
  59   * @brief Decode a base64 encoded string into plaintext (returns a buffer)
  60   * @param src A c-style string to decode
  61   * @param len Reference to length of string (to be updated on return)
  62   * @return A decoded NULL-terminated c-style string (must be free()d later)
  63   */
  64 char *b64dec(const unsigned char *data, size_t *len);
  65 
  66 /**
  67  * @brief Base64 decode a string
  68  * @param string The string to decode
  69  * @return A new, decoded string
  70  */
  71 String base64Decode(const String&);
  72 
  73 /**
  74   * @brief Decode a base64 encoded string into plaintext (using a given buffer)
  75   * @param data The c-style string to decode
  76   * @param len Reference to length of string (to be updated on return)
  77   * @param dest Reference to the buffer to decode into
  78   */
  79 void b64dec_buf(const unsigned char *data, size_t *len, char *dest);
  80 
  81 BDLIB_NS_END
  82 #endif /* !_BD_BASE64_H */
