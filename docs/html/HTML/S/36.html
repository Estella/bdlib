  25 #define _BD_SORTEDLIST_H 1

  27 #include "bdlib.h"
  28 #include "List.h"

  30 BDLIB_NS_BEGIN
  31 
  32 template <class T>
  33 /**
  34  * @class SortedList
  35  * @brief A List which sorts on insert
  36  */
  37 class SortedList : public List<T> {
  38   typedef T iterator_type;
  39   typedef typename List<iterator_type>::Node Node;
  40 
  41   public:
  42     SortedList() : List<iterator_type>() {};
  43     virtual ~SortedList() {};
  44 
  45     /**
  46      * @brief Insert an element into the list (ordered)
  47      * @param ptr The element to insert
  48      * This operation is O(n/2)
  49      */
  50     virtual void insert(const iterator_type& ptr) {
  51       /*
  52        * 1) list = NULL
  53        * 2) list = (X)
  54        * 3) list = (X)->(Y)
  55        * 4) list = (W)->(X)->(Y)
  56        */
  57       Node* node = new Node(ptr), *prev = NULL, *current = this->head;
  58       while (current != NULL && ptr < current->ptr) {
  59         prev = current;
  60         current = current->next;
  61       }
  62 
  63       if (current == NULL && this->head == NULL) {
  64         this->head = this->tail = node;
  65       } else if (current == NULL) { //Ran off the end, insert AS tail
  66         this->tail = prev->next = node;
  67         node->prev = prev; //prev=tail
  68       } else if (current == this->head) {
  69         node->next = this->head;
  70         current->prev = node;
  71         this->head = node;
  72       } else {
  73         node->next = current;
  74         node->prev = prev;
  75         current->prev = node;
  76         prev->next = node;
  77       }
  78 
  79       ++(this->my_size);
  80     }
  81 
  82 };
  83 
  84 BDLIB_NS_END
  85 #endif /* !_BD_SORTEDLIST_H */
