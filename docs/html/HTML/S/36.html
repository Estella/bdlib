  23 #ifndef _BD_BASE64_H
  24 #define _BD_BASE64_H 1
  25 
  26 #include "bdlib.h"
  27 
  28 BDLIB_NS_BEGIN
  29 class String;
  30 
  31 class Base64 {
  32 };
  33 
  34 /**
  35   * @brief Encode a plaintext string into base64 (returns a buffer)
  36   * @param src A c-style string to encode
  37   * @param len Reference to length of string (to be updated on return)
  38   * @param charset The charset to use.
  39   * @return An encoded NULL-terminated c-style string (must be free()d later)
  40   */
  41 char *b64enc(const unsigned char *src, size_t *len, const char* charset = NULL);
  42 
  43 /**
  44  * @brief Base64 encode a string
  45  * @param string The string to encode
  46  * @param charset Optional charset to use. None will use standard base64
  47  * @return A new, encoded string
  48  */
  49 String base64Encode(const String& string, const char* charset = NULL);
  50 
  51 /**
  52   * @brief Encode a plaintext string into base64 (using a given buffer)
  53   * @param data The c-style string to encode
  54   * @param len Reference to length of string (to be updated on return)
  55   * @param dest Reference to the buffer to encode into
  56   * @param charset The charset to use.
  57   */
  58 void b64enc_buf(const unsigned char *data, size_t *len, char *dest, const char* charset);
  59 
  60 /**
  61   * @brief Decode a base64 encoded string into plaintext (returns a buffer)
  62   * @param data A c-style string to decode
  63   * @param len Reference to length of string (to be updated on return)
  64   * @param charset The charset to use.
  65   * @return A decoded NULL-terminated c-style string (must be free()d later)
  66   */
  67 char *b64dec(const unsigned char *data, size_t *len, const char* charset = NULL);
  68 
  69 /**
  70  * @brief Base64 decode a string
  71  * @param string The string to decode
  72  * @param charset Optional charset to use. None will use standard base64
  73  * @return A new, decoded string
  74  */
  75 String base64Decode(const String& string, const char* charset = NULL);
  76 
  77 /**
  78   * @brief Decode a base64 encoded string into plaintext (using a given buffer)
  79   * @param data The c-style string to decode
  80   * @param len Reference to length of string (to be updated on return)
  81   * @param dest Reference to the buffer to decode into
  82   * @param charset_index The character set to use
  83   */
  84 void b64dec_buf(const unsigned char *data, size_t *len, char *dest, const char* charset_index);
  85 
  86 BDLIB_NS_END
  87 #endif /* _BD_BASE64_H */
