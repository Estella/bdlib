  36 template <class T>
  37 /**
  38  * @class Queue
  39  * @brief A FIFO Queue, what else!
  40  */
  41 class Queue : public List<T> {
  42   typedef typename List<T>::Node Node;
  43 
  44   public:
  45     friend class QueueIterator<T>;
  46     friend class QueueConstIterator<T>;
  47 
  48     typedef QueueIterator<T> iterator;
  49     typedef QueueConstIterator<T> const_iterator;
  50     typedef T value_type;
  51 
  52     typedef size_t             size_type;
  53     typedef ptrdiff_t          difference_type;
  54     typedef value_type*        pointer;
  55     typedef const value_type*  const_pointer;
  56     typedef value_type&        reference;
  57     typedef const value_type&  const_reference;
  58 
  59 
  60     Queue() : List<value_type>() {};
  61     virtual ~Queue() {};
  62 
  63     /**
  64      * @brief Dequeue an element from the tail
  65      */
  66     value_type dequeue() {
  67       value_type item = this->tail->ptr;
  68       this->deleteNode(this->tail);
  69       return item;
  70     };
  71 
  72     /**
  73      * @sa dequeue
  74      */
  75     inline friend Queue<value_type>& operator>> (Queue<value_type>& queue, reference item) {
  76       item = queue.dequeue();
  77       return queue;
  78     }
  79 
  80     /**
  81      * @brief Return the first element, but don't dequeue it
  82      */
  83     inline const_iterator front() {
  84       return this->isEmpty() ? this->end() : const_iterator(this, this->tail);
  85     }
  86 
  87 
  88     /*
  89      * @brief Enqueue an element into the queue
  90      * @param item The element to enqueue
  91      * @todo Make this take an iterator as a position to insert at
  92      */
  93     inline void enqueue(const_reference item) {
  94       this->insert(item);
  95     };
  96 
  97     /**
  98      * @sa push
  99      */
 100     inline friend Queue<value_type>& operator<< (Queue<value_type>& queue, const_reference item) {
 101       queue.enqueue(item);
 102       return queue;
 103     }
 104 
 105     inline iterator begin() { return iterator(this, this->tail); };
 106     inline iterator end() { return iterator(this, NULL); };
 107     inline const_iterator begin() const { return const_iterator(this, this->tail); };
 108     inline const_iterator end() const { return const_iterator(this, NULL, 1); };
 109 };

 111 template <class T>
 112 class QueueIterator {
 113   friend class Queue<T>;
 114   friend class QueueConstIterator<T>;
 115 
 116   public:
 117     typedef Queue<T> Queue_t;
 118     typedef QueueIterator<T> iterator;
 119     typedef QueueConstIterator<T> const_iterator;
 120     typedef typename Queue_t::Node Node;
 121     typedef std::forward_iterator_tag iterator_category;
 122     typedef T value_type;
 123     typedef ptrdiff_t difference_type;
 124     typedef size_t size_type;
 125     typedef T& reference;
 126     typedef T* pointer;
 127 
 128   private:
 129     Queue_t *queue;
 130     Node *current;
 131 
 132     QueueIterator(Queue_t* _queue, Node* start) : queue(_queue), current(start) {};
 133   public:
 134     QueueIterator() : queue(NULL), current(NULL) {};
 135     ~QueueIterator() { }
 136 
 137     reference operator *() const { return current->ptr; }
 138     pointer operator ->() const { return &(operator*()); }
 139 
 140 
 141     bool operator==(const iterator& iter) const { return iter.current == current; }
 142     bool operator!=(const iterator& iter) const { return iter.current != current; }
 143 
 144     //Postfix
 145     iterator operator ++(int) {
 146       iterator tmp(*this);
 147       current = current->next;
 148       return tmp;
 149     }
 150 
 151     iterator& operator ++() {
 152       current = current->next;
 153       return *this;
 154     }
 155 };
