  34 template <class T>
  35 /**
  36  * @class StringBuf
  37  * @brief Helps the String class with reference counting
  38  * @todo look into something like a string object made of string pointers, and << displays all of the pointed to objects. (this would save copying to construct immuatable concatenated strings, but would only work with the << stream
  39  */
  40 class ArrayRef {
  41   public:
  42     ~ArrayRef() { FreeBuf(buf); };
  43     mutable size_t len; //Length of string
  44     mutable size_t size; //Capacity of buffer
  45     mutable T* buf;
  46     mutable uint8_t n; //References
  47 
  48     ArrayRef() : len(0), size(0), buf(NULL), n(1) {};
  49     /**
  50      * @brief Ensure that the buffer capacity() is >= newSize; else grow/copy into larger buffer.
  51      * @param newSize A size that we need to Allocate the buffer to.
  52      * @pre newSize is > 0 (assumed as size_t is unsigned)
  53      * @post The buffer is at least nsize bytes long.
  54      * @post If the buffer had to grow, the old data was deep copied into the new buffer and the old deleted.
  55      */
  56     void Reserve(const size_t newSize, size_t& offset) const
  57     {
  58       /* Don't new if we already have enough room! */
  59       if (size < newSize) {
  60         size = std::max(size_t(size * 1.5), newSize);
  61 
  62         T *newbuf = AllocBuf(size);
  63 
  64         if (newbuf != buf) {
  65           /* Copy old buffer into new - only copy the substring */
  66           std::copy(buf + offset, buf + offset + len, newbuf);
  67           FreeBuf(buf);
  68           buf = newbuf;
  69           offset = 0;
  70         }
  71       } else if ((size - offset) < newSize) {
  72         // There's enough room in the current buffer, but we're offsetted/shifted to a point where there's no room left
  73         // Shift everything to the beginning and reset the offset
  74         /* Only copy the substring */
  75         memmove(buf, buf + offset, len);
  76         offset = 0;
  77       }
  78     }
  79     //void Reserve(const size_t, size_t&) const;
  80 
  81     /**
  82      * @brief Allocates a buffer and returns it's address.
  83      * @param bytes The number of bytes to allocate.
  84      * @post A new block of memory is allocated.
  85      * @todo Implement mempool here.
  86      */
  87     inline T* AllocBuf(const size_t bytes) const {
  88       return new T[bytes];
  89     }
  90 
  91     /**
  92      * @brief Free's up the allocated buffer.
  93      * @param p The buffer to be free'd
  94      * @post The buffer is deleted.
  95      * @todo Implement mempool here.
  96      */
  97     inline void FreeBuf(T* p) const {
  98       delete[] p;
  99     }
 100 
 101     /**
 102      * @brief Is this string shared?
 103      */
 104     inline bool isShared() const { return n > 1; };
 105   private:
 106     // No copying allowed
 107     ArrayRef(const ArrayRef&); ///<Block implicit copy constructor
 108     ArrayRef& operator=(const ArrayRef&); ///<Block implicit copy constructor
 109 };

 159 template <class T>
 160 /**
 161  * @class ReferenceCountedArray
 162  * @brief
 163  */
 164 class ReferenceCountedArray {
 165   public:
 166     typedef T value_type;
 167 
 168     typedef size_t             size_type;
 169     typedef ptrdiff_t          difference_type;
 170     typedef value_type*        pointer;
 171     typedef const value_type*  const_pointer;
 172     typedef value_type&        reference;
 173     typedef const value_type&  const_reference;
 174 
 175   protected:
 176     /**
 177      * @brief Detach from the shared reference.
 178      * This is only called when losing the old buffer or when modifying the buffer (and copy-on-write is used)
 179      * @note This does not free the old reference, as it is still in use
 180      */
 181     void doDetach() const {
 182       decRef();
 183       Ref = new ArrayRef<value_type>();
 184       sublen = 0;
 185       offset = 0;
 186     }
 187 
 188     /**
 189      * @brief Set the lengths to the specified length
 190      * @param newLen the new length to set to
 191      */
 192     inline void setLength(size_t newLen) const { Ref->len = sublen = newLen; };
 193 
 194     /**
 195      * @sa setLength()
 196      */
 197     inline void addLength(size_t diff) const { Ref->len += diff; sublen += diff; };
 198 
 199     /**
 200      * @sa setLength()
 201      */
 202     inline void subLength(size_t diff) const { Ref->len -= diff; sublen -= diff; };
 203 
 204     /**
 205      * @brief Mutable Ref->buf+offset reference for use internally
 206      */
 207     //pointer mdata() const { return Buf() + offset; };
 208 
 209     /**
 210      * @brief Mutable Ref->buf reference for use internally
 211      */
 212     inline pointer Buf(int i = 0) const { return &Ref->buf[offset + i]; };
 213 
 214     /**
 215      * @brief Ref->buf reference for use internally
 216      */
 217     inline const_pointer constBuf(int i = 0) const { return Buf(i); };
 218 
 219     /**
 220      * @brief Increment our reference counter.
 221      */
 222     inline uint8_t incRef() const { return ++Ref->n; };
 223 
 224     /**
 225      * @brief Decrement our reference counter.
 226      */
 227     inline uint8_t decRef() const { return --Ref->n; };
 228 
 229     /**
 230      * @return True if this string is shared; false if not.
 231      */
 232     inline bool isShared() const { return Ref->isShared(); };
 233   protected:
 234     /**
 235      * @brief The array reference for reference counting
 236      * This is mutable so that Ref->n can be modified, which really is mutable
 237      */
 238     mutable ArrayRef<value_type> *Ref;
 239 
 240     /**
 241      * This is for subarrays: so we know where the subarray starts.
 242      */
 243     mutable size_t offset;
 244     /**
 245      * This is for subarrays: so we know where the subarray ends.
 246      */
 247     mutable size_t sublen;
 248 
 249     /**
 250      * @brief Free up our reference if we have the last one.
 251      * @post The reference counter is decremented.
 252      * @post If this was the last Reference, it is free'd
 253      * This is only called in ~Array() and operator=(Array&).
 254      * It checks whether of not this Array was the last reference to the buffer, and if it was, it removes it.
 255      */
 256     inline void CheckDeallocRef() {
 257       if (decRef() < 1)
 258         delete Ref;
 259     }
 260 
 261     /**
 262      * @brief Detach from the reference
 263      * This is called when the old buffer is no longer needed for this Array.
 264      * ie, operator=() was called.
 265      */
 266     void Detach() {
 267       if (isShared()) {
 268         doDetach();
 269       } else {
 270         setLength(0);
 271         offset = 0;
 272       }
 273     }
 274 
 275     //    void COW(size_t) const;
 276     /**
 277      * @brief Ensure that our internal buffer is unshared.
 278      * @param n Create/Grow the buffer to this size.
 279      * @pre n is > 0; this is assumed due to size_t being unsigned though.
 280      * @post The internal buffer/data is unshared
 281      * @post The buffer is at least size n.
 282      * @post The buffer is deep copied to a new buffer.
 283      *
 284      * Ensure that our internal buffer is unshared.
 285      * If needed, performs a deep copy into a new buffer (COW).
 286      * Also take a hint size n of the new string's size as to avoid needless copying/allocing.
 287      */
 288     void COW(size_t n) const {
 289       const_pointer oldBuf = constBuf();
 290       size_t oldLength = length();
 291       size_t oldCapacity = capacity();
 292 
 293       doDetach(); //Detach from the shared reference
 294       Reserve( std::max(oldCapacity, n) ); //Will set capacity()/size
 295       std::copy(oldBuf, oldBuf + oldLength, Buf());
 296       setLength(oldLength);
 297     }
 298 
 299     inline void AboutToModify(size_t n) const {
 300       if (isShared())
 301         COW(n); // Clears the offset
 302       else {
 303         Reserve(n);
 304         /* Shift the offset away */
 305       }
 306     }
 307     inline void getOwnCopy() const { AboutToModify(capacity()); };
 308   public:
 309     ReferenceCountedArray() : Ref(new ArrayRef<value_type>()), offset(0), sublen(0) {};
 310     ReferenceCountedArray(const ReferenceCountedArray& rca) : Ref(rca.Ref), offset(rca.offset), sublen(rca.sublen) { incRef(); };
 311     /**
 312      * @brief Create an empty String container with at least the specified bytes in size.
 313      * @param newSize Reserve at least this many bytes for this String.
 314      * @post This string's memory will also never be shrunk.
 315      * @post A buffer has been created.
 316      *
 317      * The idea behind this is that if a specific size was asked for, the buffer is like
 318      * a char buf[N];
 319      */
 320     explicit ReferenceCountedArray(const size_t newSize) : Ref(new ArrayRef<value_type>()), offset(0), sublen(0) {
 321       if (newSize <= 0) return;
 322       Reserve(newSize);
 323     };
 324 
 325     /**
 326      * @brief Array Destructor
 327      * @post If the Array's Reference is not shared, it is free'd.
 328      * @post If the Array's Reference IS shared, it is decremented and detached.
 329      */
 330     virtual ~ReferenceCountedArray() { CheckDeallocRef(); };
 331 
 332     /**
 333      * @brief Sets our Reference to the given String reference.
 334      * @param rca The String object to reference.
 335      * @post The old buffer (if we had one) is free'd.
 336      * @post Our Reference now points to the given String.
 337      * @post Our old rca object has been deleted (disconnected).
 338      * @return The new rca object.
 339      * This handles self-assignment just fine, checking for it explicitly would be ineffecient for most cases.
 340      */
 341     const ReferenceCountedArray& operator=(const ReferenceCountedArray &rca) {
 342       rca.incRef();
 343       offset = rca.offset;
 344       sublen = rca.sublen;
 345       CheckDeallocRef();
 346       Ref = rca.Ref;
 347       return *this;
 348     }

 350     /**
 351      * @brief Sets our buffer to the given item.
 352      * @param item The item to set our buffer to.
 353      * @post The old buffer (if we had one) is free'd.
 354      * @post A sufficiently sized new buffer is made with the item within.
 355      * @return The new array object.
 356      */
 357     const ReferenceCountedArray& operator=(const_reference item) {
 358       Detach();
 359       append(item);
 360       return *this;
 361     }

 406     /**
 407      * @brief Data accessor
 408      * @return Pointer to array of characters (not necesarily null-terminated).

 416     /*
 417      * @brief Find an item in the array
 418      * @return The position of the item if found, or npos if not found
 419      **/
 420     size_t find(const_reference item) const {
 421       for (size_t i = 0; i < length(); ++i)
 422         if (*(Buf(i)) == item)
 423           return i;
 424       return size_t(-1);

 432      */
 433     inline bool hasIndex(int i) const {
 434 #ifdef DEBUG
 435       if (i < 0 || i >= (int) (offset + length())) ::printf("ATTEMPT TO ACCESS INDEX %d/%zu\n", i, size_t(offset + length()));

 437       return (i < (int) length());

 442      * Unlinke at() this is unchecked.

 444     inline value_type read(int i) const { return *(constBuf(i)); };

 457     /**
 458      * @class Cref
 459      * @brief Safe element reading and writing.
 460      * @todo This should not provide copy constructors for Cref, they shouldn't be needed because of const char String::operator[]
 461      * This class should be optimized away and fully inlined such that:
 462      * String s;
 463      * s[0] = 'a';
 464      * Should be rewritten as:
 465      * s.write(0, 'a');
 466      */
 467     class Cref {
 468       private:
 469         friend class ReferenceCountedArray;
 470         ReferenceCountedArray& rca;
 471         int k;
 472 
 473         /**
 474          * @brief Used by String::Cref operator[]
 475          */
 476         Cref(ReferenceCountedArray& _rca, int i) : rca(_rca), k(i) {};
 477         Cref(); //Not defined - never used
 478 
 479       public:
 480         Cref(const Cref& cref) : rca(cref.rca), k(cref.k) {};
 481         inline Cref& operator=(const Cref& cref) {
 482           (*this) = value_type(cref);
 483           return (*this);
 484         }
 485 
 486         public:
 487         /**
 488          * @sa ReferenceCountedArray::operator[]
 489          */
 490         inline operator value_type() const { return rca.read(k); };
 491 
 492         /**
 493          * Stroustrup shows using this as void with no return value, but that breaks chaining a[n] = b[n] = 'b';
 494          */
 495         inline Cref& operator=(value_type c) {
 496           rca.write(k, c);
 497           return (*this);
 498         };
 499     };

 507     /**
 508      * @brief Returns the character at the given index.
 509      * @return The character at the given index.
 510      * @param i Index to return.
 511      * @pre The index must exist.
 512      * @sa operator[]()
 513      * @todo Perhaps this should throw an exception if out of range?
 514      */
 515     inline value_type at(int i) const { return hasIndex(i) ? (*this)[i] : 0; };
 516 
 517     /**
 518      * @brief Return a new array from a subarray
 519      * @return a new ReferenceCountedArray
 520      * @param start The offset to begin the subarray from (indexed from 0)
 521      * @param len The length of the subarray to return
 522      * The returned slice is a reference to the original array until modified.
 523      */
 524     void slice(int start, int len) {
 525       // Start is after the end, set us to an empty array
 526       if (start >= (signed) length()) {
 527         offset = length();
 528         setLength(0);
 529         return;
 530       }
 531 
 532       //Count backwards from the end
 533       if (start < 0)
 534         start = length() + start;
 535       // Start was before the beginning, just reset to the beginning
 536       if (start < 0)
 537         start = 0;
 538 
 539       size_t newOffset = offset + start;
 540       offset = newOffset;
 541       //If the length of the subarray exceeds the end of the array, truncate to the end of the array
 542       if (start + len >= (signed) length())
 543         len = length() - start;
 544       // If the length is negative, stop from counting backwards from the end
 545       else if (len < 0)
 546         len = length() - start + len;
 547       setLength(len);
 548     }

 568     /**
 569      * @brief Inserts a ReferenceCountedArray object into our buffer
 570      * @param k The index to insert at.
 571      * @param rca The rca to insert.
 572      * @param n The length to insert.
 573      * @post The buffer contains n items from rca inserted at index k.
 574      */
 575     void insert(int k, const ReferenceCountedArray& rca, int n = -1) {
 576       if (n == 0) return;
 577       if (k && !hasIndex(k-1)) return;
 578 
 579       int slen = rca.length();
 580 
 581       /* New rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 582       if (k == 0 && size_t(slen) > length() && (n == -1 || n == slen)) {
 583         *this = rca;
 584         return;
 585       }
 586 
 587       if (n == -1 || n > slen)
 588         n = slen;
 589       slen -= slen - n;
 590       AboutToModify(length() + slen);
 591       memmove(Buf() + k + slen, Buf() + k, length() - k);
 592       std::copy(rca.begin(), rca.begin() + slen, Buf() + k);
 593       addLength(slen);
 594     }
 595 
 596     /**
 597      * @brief Insert an item at the given index.
 598      * @param k The index to insert at.
 599      * @param item The item to be inserted.
 600      * @post A buffer is allocated.
 601      * @post If the old buffer was too small, it is enlarged.
 602      * @post The item is inserted at the given index.
 603      */
 604     void insert(int k, const_reference item)
 605     {
 606       if (k && !hasIndex(k-1)) return;
 607 
 608       AboutToModify(length() + 1);
 609       memmove(Buf() + k + 1, Buf() + k, length() - k);
 610       *(Buf(k)) = item;
 611       addLength(1);
 612     }
 613 
 614     /**
 615      * @brief Replace the given index with the given item.
 616      * @param k The index to replace.
 617      * @param item The item to replace with.
 618      * @post The given index has been replaced.
 619      * @post COW is done if needed.
 620      */
 621     void replace(int k, const_reference item) {
 622       if (k && !hasIndex(k-1)) return;
 623 
 624       getOwnCopy();
 625       *(Buf(k)) = item;
 626     }
 627 
 628     /**
 629      * @brief Replaces n elements in our buffer at index k with the given ReferenceCountedArray object
 630      * @param k The index to replace at.
 631      * @param rca The ReferenceCountedArray object to replace with.
 632      * @param n The number of characters to use for the replace.
 633      */
 634     void replace(int k, const ReferenceCountedArray &rca, int n = -1) {
 635       if (n == 0) return;
 636       if (k && !hasIndex(k-1)) return;
 637 
 638       int slen = rca.length();
 639 
 640       /* Replace rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 641       if (k == 0 && size_t(slen) > length() && (n == -1 || n == slen)) {
 642         *this = rca;
 643         return;
 644       }
 645 
 646       if (n == -1 || n > slen)
 647         n = slen;
 648       slen -= slen - n;
 649 
 650       size_t newlen = k + slen;
 651 
 652       if (newlen >= length()) {
 653         AboutToModify(newlen);
 654       } else {
 655         newlen = length();
 656         getOwnCopy();
 657       }
 658       std::copy(rca.begin(), rca.begin() + slen, Buf() + k);
 659       setLength(newlen);
 660     }
