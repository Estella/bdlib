  69 void b64enc_buf(const unsigned char *data, size_t *len, char *dest)
  70 {
  71   char *buf = dest;
  72 
  73   /* Encode 3 bytes at a time. */
  74 
  75   /*
  76    *
  77    * |       0       |       1       |       2       |
  78    *
  79    * |               |               |               |
  80    * |       |       |       |       |       |       |
  81    * |   |   |   |   |   |   |   |   |   |   |   |   |
  82    * | | | | | | | | | | | | | | | | | | | | | | | | |
  83    *  7 6 5 4 3 2 1 0 7 6 5 4 3 2 1 0 7 6 5 4 3 2 1 0
  84    * -------------------------------------------------
  85    *  5 4 3 2 1 0 5 4 3 2 1 0 5 4 3 2 1 0 5 4 3 2 1 0 
  86    * | | | | | | | | | | | | | | | | | | | | | | | | |
  87    * |   | | |   |   | | |   |   | | |   |   | | |   |
  88    * |     |     |     |     |     |     |     |     |
  89    * |           |           |           |           |
  90    *
  91    * |     0     |     1     |     2     |     3     |
  92    *
  93    */
  94 
  95 
  96   while (*len >= NUM_ASCII_BYTES) {
  97     /* buf[0] is the 6 left-most bits of data[0] */
  98     buf[0] = b64_charset[(data[0] & 0xfc) >> 2];
  99     /* buf[1] is the right-most 2 bits of data[0] and the left-most 4 bits of data[1] */
 100     buf[1] = b64_charset[((data[0] & 0x03) << 4) | ((data[1] & 0xf0) >> 4)];
 101     /* buf[2] is the right-most 4 bits of data[1] and the 2 left-most bits of data[2] */
 102     buf[2] = b64_charset[((data[1] & 0x0f) << 2) | ((data[2] & 0xc0) >> 6)];
 103     /* buf[3] is the right-most 6 bits of data[2] */
 104     buf[3] = b64_charset[data[2] & 0x3f];
 105 
 106     data += NUM_ASCII_BYTES;
 107     buf += NUM_ENCODED_BYTES;
 108     *len -= NUM_ASCII_BYTES;
 109   }
 110 
 111   /* There is either 1 or 2 bytes left to encode (and possibly some padding to add) */
 112   if (*len > 0) {
 113     buf[0] = b64_charset[(data[0] & 0xfc) >> 2];
 114     if (*len == 1) {
 115       buf[1] = b64_charset[((data[0] & 0x03) << 4)];
 116       buf[2] = PADDING_CHAR;
 117     } else if (*len > 1) { /* == 2 .. left as >1 to help optimize with last if */
 118       /* Tack on to buf[1] the left-most 4 bits of data[1] */
 119       buf[1] = b64_charset[((data[0] & 0x03) << 4) | ((data[1] & 0xf0) >> 4)];
 120       buf[2] = b64_charset[(data[1] & 0x0f) << 2];
 121     }
 122     buf[3] = PADDING_CHAR;
 123     buf += NUM_ENCODED_BYTES;
 124   }
 125 
 126   *len = buf - dest;
 127 }

 129 String base64Encode(const String& string) {

 132   bd::String encoded(p, len);
 133   free(p);

 137 char *b64enc(const unsigned char *src, size_t *len)
 138 {
 139   /* Take the length and round up the next 4-byte boundary */
 140   size_t dlen = (((*len + (NUM_ASCII_BYTES - 1)) / NUM_ASCII_BYTES) * NUM_ENCODED_BYTES);
 141   char *dest = (char *) malloc(dlen + 1);;
 142 
 143   b64enc_buf(src, len, dest);
 144   dest[*len] = '\0';
 145   return dest;
 146 }

 148 void b64dec_buf(const unsigned char *data, size_t *len, char *dest)
 149 {
 150   char *buf = dest;
 151   size_t blocks = (*len / NUM_ENCODED_BYTES) + ((*len % NUM_ENCODED_BYTES) ? 1 : 0);
 152 
 153   /* Convert the encoded base64 character back into our base255 ASCII character */
 154   while (blocks > 0) {
 155     /* buf[0] is the 6 bits from data[0] + left-most 2 bits of data[1] */
 156     *buf++ = (b64_indexes[data[0]] << 2) + ((b64_indexes[data[1]] & 0x30) >> 4);  /* 0x30 not required but to be clear. */
 157     if (data[2] != PADDING_CHAR) {
 158       /* buf[1] is the right-most 4 bits of data[1] + left-most 4 bits of data[2] */
 159       *buf++ = ((b64_indexes[data[1]] & 0x0f) << 4) + ((b64_indexes[data[2]] & 0x3c) >> 2);
 160       if (data[3] != PADDING_CHAR) {
 161         /* buf[2] is the right-most 2 bits of data[2] + data[3] */
 162         *buf++ = ((b64_indexes[data[2]] & 0x03) << 6) + b64_indexes[data[3]];
 163       }
 164     }
 165     data += NUM_ENCODED_BYTES;
 166     --blocks;
 167   }
 168 
 169   *len = (buf - dest);
 170 }

 172 char *b64dec(const unsigned char *data, size_t *len)
 173 {
 174   size_t dlen = ((*len / NUM_ENCODED_BYTES) + ((*len % NUM_ENCODED_BYTES) ? 1 : 0)) * NUM_ASCII_BYTES;
 175   char *dest = (char *) malloc(dlen + 1);
 176 
 177   b64dec_buf(data, len, dest);
 178   dest[*len] = '\0';
 179   return dest;
 180 }
