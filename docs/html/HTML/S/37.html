  41 /**
  42  * @brief Compare our String object with another String object, but only n characters
  43  * @param str The String object to compare to.
  44  * @param n The number of characters to compare.
  45  * @return an integer less than, equal to, or greater than zero if our buffer is found, respectively, to be less than, to match, or be greater than str.
  46 */
  47 int String::compare(const String &str, size_t n) const
  48 {
  49   size_t my_len = length();
  50   size_t slen = std::min(str.length(), n);
  51   size_t len = std::min(my_len, slen);
  52 
  53   /* Same string? */
  54   if (data() == str.data() && my_len == str.length())
  55     return 0;
  56 
  57   for (size_t i = 0; i < len; ++i) {
  58     if ((*this)[i] < str[i])
  59       return -1;
  60     else if (str[i] < (*this)[i])
  61       return 1;
  62   }
  63   return my_len - slen;
  64 }

  68 /**
  69  * @brief Insert a cstring at the given index.
  70  * @param string The cstring to be inserted.
  71  * @param k The index to insert at.
  72  * @param n Up to this many characters will be used from the string.
  73  * @post A buffer is allocated.
  74  * @post If the old buffer was too small, it is enlarged.
  75  * @post The string is inserted at the given index.
  76  */
  77 void String::insert(int k, const char *string, int n)
  78 {
  79   if (n == 0) return;
  80   if (k && !hasIndex(k-1)) return;
  81   
  82   size_t slen = (n == -1) ? strlen(string) : (size_t) n;
  83 
  84   AboutToModify(length() + slen);
  85   memmove(Buf() + k + slen, Buf() + k, length() - k);
  86   std::copy(string, string + slen, Buf() + k);
  87   addLength(slen);
  88 }

  90 /**
  91  * @brief Replaces the given index with the given cstring up to the given amount of characters.
  92  * @param k The index to replace at.
  93  * @param string The cstring to replace with.
  94  * @param n How many characters to use from string.
  95  */
  96 void String::replace(int k, const char *string, int n)
  97 {
  98   if (n == 0) return;
  99   if (k && !hasIndex(k-1)) return;
 100 
 101   size_t slen = (n == -1) ? strlen(string) : (size_t) n;
 102   size_t newlen = k + slen;
 103   
 104   if (newlen >= length()) {
 105     AboutToModify(newlen);
 106   } else {
 107     newlen = length();
 108     getOwnCopy();
 109   }
 110   std::copy(string, string + slen, Buf() + k);
 111   setLength(newlen);
 112 }

 114 std::istream& operator >> (std::istream& is, String &string) {
 115   char ch;
 116   string = "";    // empty string, will build one char at-a-time
 117   is >> ch;    // whitespace skipped, first non-white char in ch
 118 
 119   if (!is.fail())
 120   {
 121     do
 122     {
 123       string += ch;
 124       is.get(ch);
 125     } while (!is.fail() && !isspace(ch));
 126 
 127     if (isspace(ch))    // put whitespace back on the stream
 128     {
 129       is.putback(ch);
 130     }
 131   }
 132   return is;
 133 }

 146 /*
 147  * @brief Split a string by a delimiter
 148  * @param delim The delimiter to use
 149  * @param limit Optional param which specified the max number of elements to return
 150  * @note Excess whitespace is trimmed similar to Ruby's split.
 151  */
 152 Array<String> String::split(const String& delim, size_t limit) const {
 153   if (!length()) return Array<String>();
 154   if (limit == 1) {
 155     String list[] = {*this};
 156     return Array<String>(list, 1);
 157   }
 158 
 159   Array<String> array;
 160   // Use a temporary - is a fast reference and never modified, so no COW ever used.
 161   String str(*this);
 162   size_t pos = 0;
 163 
 164   while (array.size() < limit - 1) {
 165     // Trim out left whitespace
 166     if (delim == ' ') while (str.length() && str[0] == ' ') ++str;
 167 
 168     // All done when there's nothing left
 169     if (!str) break;
 170 
 171     if ((pos = str.find(delim)) == npos)
 172       pos = str.length();
 173     array << str(0, pos);
 174     str += int(pos + delim.length());
 175   }
 176 
 177   // Add on extra
 178   if (limit != npos) {
 179     // Trim out left whitespace
 180     if (delim == ' ') while (str.length() && str[0] == ' ') ++str;
 181     array << str;
 182   }
 183 
 184   return array;
 185 }

 197   String newString((this->length() * times));

 199     newString += *this;

 214 String& String::chomp() {
 215   if (length() && (*this)[length() - 1] == '\n') --(*this);
 216   if (length() && (*this)[length() - 1] == '\r') --(*this);
 217   return *this;
 218 }

 220 String& String::trim() {
 221   // ltrim
 222   while (length() && isspace((*this)[0])) { ++offset; subLength(1); }
 223   // rtrim
 224   while (length() && isspace((*this)[length() - 1])) subLength(1);
 225   return *this;

 228 size_t String::find (const String& str) const {
 229   for (const char* c = begin(); c != end(); ++c)
 230     if (strncmp(c, str.c_str(), std::min(str.length(), size_t(end() - c))) == 0)
 231       return size_t(c - begin());
 232   return npos;
 233 }

 235 /**
 236  * @brief DJB's hash function
 237  */
 238 size_t String::hash() const {
 239   size_t _hash = 5381;
 240 
 241   for(size_t i = 0; i < length(); ++i)
 242     _hash = ((_hash << 5) + _hash) + data()[i];
 243   return (_hash & 0x7FFFFFFF);
 244 }

 246 /**
 247  * @brief Splits off tokens
 248  * @param str The string to split a token off of
 249  * @param delim The delimiter to tokenize with
 250  * This will modify the given string to remove the first token, which is returned.
 251  * If no more tokens exist, the remaining string is returned and the string becomes "". Future calls will return a blank String("")
 252  */
 253 String newsplit(String& str, char delim)
 254 {
 255   if (!str.length()) return "";
 256   size_t pos = str.find(delim);
 257   if (pos == String::npos)
 258     pos = str.length();
 259 
 260   String first(str(0, pos));
 261 
 262   /* Trim out runs of whitespaces */
 263   if (delim == ' ') {
 264     while ((str.begin() + pos + 1) < str.end() && str[pos + 1] == ' ')
 265       ++pos;
 266   }
 267 
 268   str = str(pos + 1, str.length() - pos + 1);
 269   return first;
 270 }
