  41 /**
  42  * @brief Compare our String object with another String object, but only n characters
  43  * @param str The String object to compare to.
  44  * @param n The number of characters to compare.
  45  * @return an integer less than, equal to, or greater than zero if our buffer is found, respectively, to be less than, to match, or be greater than str.
  46 */
  47 int String::compare(const String &str, size_t n) const
  48 {
  49   size_t my_len = length();
  50   size_t slen = std::min(str.length(), n);
  51   size_t len = std::min(my_len, slen);
  52 
  53   /* Same string? */
  54   if (data() == str.data() && my_len == str.length())
  55     return 0;
  56 
  57   for (size_t i = 0; i < len; ++i) {
  58     if ((*this)[i] < str[i])
  59       return -1;
  60     else if (str[i] < (*this)[i])
  61       return 1;
  62   }
  63   return my_len - slen;
  64 }

  68 /**
  69  * @brief Insert a cstring at the given index.
  70  * @param string The cstring to be inserted.
  71  * @param k The index to insert at.
  72  * @param n Up to this many characters will be used from the string.
  73  * @post A buffer is allocated.
  74  * @post If the old buffer was too small, it is enlarged.
  75  * @post The string is inserted at the given index.
  76  */
  77 void String::insert(int k, const char *string, int n)
  78 {
  79   if (n == 0) return;
  80   if (k && !hasIndex(k-1)) return;
  81   
  82   size_t slen = (n == -1) ? strlen(string) : (size_t) n;
  83 
  84   AboutToModify(length() + slen);
  85   memmove(Buf() + k + slen, Buf() + k, length() - k);
  86   std::copy(string, string + slen, Buf() + k);
  87   addLength(slen);
  88 }

  90 /**
  91  * @brief Replaces the given index with the given cstring up to the given amount of characters.
  92  * @param k The index to replace at.
  93  * @param string The cstring to replace with.
  94  * @param n How many characters to use from string.
  95  */
  96 void String::replace(int k, const char *string, int n)
  97 {
  98   if (n == 0) return;
  99   if (k && !hasIndex(k-1)) return;
 100 
 101   size_t slen = (n == -1) ? strlen(string) : (size_t) n;
 102   size_t newlen = k + slen;
 103   
 104   if (newlen >= length()) {
 105     AboutToModify(newlen);
 106   } else {
 107     newlen = length();
 108     getOwnCopy();
 109   }
 110   std::copy(string, string + slen, Buf() + k);
 111   setLength(newlen);
 112 }

 127 std::istream& operator >> (std::istream& is, String &string) {
 128   char ch;
 129   string = "";    // empty string, will build one char at-a-time
 130   is >> ch;    // whitespace skipped, first non-white char in ch
 131 
 132   if (!is.fail())
 133   {
 134     do
 135     {
 136       string += ch;
 137       is.get(ch);
 138     } while (!is.fail() && !isspace(ch));
 139 
 140     if (isspace(ch))    // put whitespace back on the stream
 141     {
 142       is.putback(ch);
 143     }
 144   }
 145   return is;
 146 }

 159 /*
 160  * @brief Split a string by a delimiter
 161  * @param delim The delimiter to use
 162  * @param limit Optional param which specified the max number of elements to return
 163  * @note Excess whitespace is trimmed similar to Ruby's split.
 164  */
 165 Array<String> String::split(const String& delim, size_t limit) const {
 166   if (!length()) return Array<String>();
 167   if (limit == 1) {
 168     String list[] = {*this};
 169     return Array<String>(list, 1);
 170   }
 171 
 172   Array<String> array;
 173   // Use a temporary - is a fast reference and never modified, so no COW ever used.
 174   String str(*this);
 175   size_t pos = 0;
 176 
 177   while (array.size() < limit - 1) {
 178     // Trim out left whitespace
 179     if (delim == ' ') while (str.length() && str[0] == ' ') ++str;
 180 
 181     // All done when there's nothing left
 182     if (!str) break;
 183 
 184     if ((pos = str.find(delim)) == npos)
 185       pos = str.length();
 186     array << str(0, pos);
 187     str += int(pos + delim.length());
 188   }
 189 
 190   // Add on extra
 191   if (limit != npos) {
 192     // Trim out left whitespace
 193     if (delim == ' ') while (str.length() && str[0] == ' ') ++str;
 194     array << str;
 195   }
 196 
 197   return array;
 198 }

 227 String& String::chomp() {
 228   if (length() && (*this)[length() - 1] == '\n') --(*this);
 229   if (length() && (*this)[length() - 1] == '\r') --(*this);
 230   return *this;
 231 }

 233 String& String::trim() {
 234   // ltrim
 235   while (length() && isspace((*this)[0])) { ++offset; subLength(1); }
 236   // rtrim
 237   while (length() && isspace((*this)[length() - 1])) subLength(1);
 238   return *this;

 241 size_t String::find (const String& str) const {
 242   for (const char* c = begin(); c != end(); ++c)
 243     if (strncmp(c, str.c_str(), std::min(str.length(), size_t(end() - c))) == 0)
 244       return size_t(c - begin());
 245   return npos;
 246 }

 248 /**
 249  * @brief DJB's hash function
 250  */
 251 size_t String::hash() const {
 252   size_t _hash = 5381;
 253 
 254   for(size_t i = 0; i < length(); ++i)
 255     _hash = ((_hash << 5) + _hash) + data()[i];
 256   return (_hash & 0x7FFFFFFF);
 257 }

 259 /**
 260  * @brief Splits off tokens
 261  * @param str The string to split a token off of
 262  * @param delim The delimiter to tokenize with
 263  * This will modify the given string to remove the first token, which is returned.
 264  * If no more tokens exist, the remaining string is returned and the string becomes "". Future calls will return a blank String("")
 265  */
 266 String newsplit(String& str, char delim)
 267 {
 268   if (!str.length()) return "";
 269   size_t pos = str.find(delim);
 270   if (pos == String::npos)
 271     pos = str.length();
 272 
 273   String first(str(0, pos));
 274 
 275   /* Trim out runs of whitespaces */
 276   if (delim == ' ') {
 277     while ((str.begin() + pos + 1) < str.end() && str[pos + 1] == ' ')
 278       ++pos;
 279   }
 280 
 281   str = str(pos + 1, str.length() - pos + 1);
 282   return first;
 283 }
