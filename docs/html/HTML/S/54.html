  37 #define STREAM_BLOCKSIZE        1024
  38 /* The possibilities for the third argument to `fseek'.
  39    These values should not be changed.  */
  40 #define SEEK_SET        0       /* Seek from beginning of file.  */
  41 #define SEEK_CUR        1       /* Seek from current position.  */
  42 #define SEEK_END        2       /* Seek from end of file.  */
  43 
  44 class Stream {
  45   protected:
  46         String str;
  47         unsigned int pos;
  48         bool loading;
  49 
  50   public:
  51         Stream() : str(), pos(0), loading(0) {};
  52         Stream(const Stream& stream) : str(stream.str), pos(stream.pos), loading(0) {};
  53         Stream(const String& string) : str(string), pos(0), loading(0) {};
  54         Stream(const int newSize) : str(), pos(0), loading(0) { if (newSize > 0) Reserve(newSize); };
  55         virtual ~Stream() {};
  56 
  57         void Reserve(const size_t) const;
  58 
  59         /**
  60          * @brief Returns the position of the Stream.
  61          * @return Position of the Stream.
  62         */
  63         size_t tell() const { return pos; };
  64 
  65         /**
  66          * @brief Truncates the stream at the current position.
  67         */
  68         void truncate() { str = str(0, pos); };
  69 
  70 //        operator void*() { return tell() <= length(); };
  71 
  72         int seek(int, int);
  73         void clear() { str.clear(); pos = 0; }
  74 
  75         /*
  76          * @brief Insert a string into the stream.
  77          * @note The stream pointer is advanced as well
  78          */
  79         virtual void puts (const String& string) {
  80           str.replace(tell(), string);
  81           pos += string.length();
  82         }
  83 
  84         /*
  85          * @brief Reads 1 line from the stream
  86          * @note The stream pointer is advanced as well
  87          * @param maxSize Optional param which specifies max data to pull
  88          * @sa gets
  89          */
  90         virtual String getline (size_t maxSize = 99999999) { return gets(maxSize == 99999999 ? (length() - pos) : maxSize, '\n'); }
  91 
  92         /*
  93          * @brief Reads specified number of bytes from the stream
  94          * @param maxSize How many bytes to read
  95          * @param delim What to split the read on. For example: '\n' will return 1 line.
  96          * @note The stream pointer is advanced as well
  97          */
  98         virtual String gets (size_t maxSize, char delim = 0);
  99 
 100         /*
 101          * @brief Load a file into the stream
 102          * @returns 1 on error, 0 on success.
 103          * @param fd File descriptor to read from
 104          * @note this.loading is set to 1 during this process.
 105          * @note Uses puts() for populating the stream.
 106          * @note The fd is not closed.
 107          * @sa puts
 108          */
 109         virtual int loadFile(const int);
 110 
 111         /*
 112          * @brief Load a file into the stream
 113          * @returns 1 on error, 0 on success.
 114          * @param fname File to read from
 115          * @note this.loading is set to 1 during this process.
 116          * @note Uses puts() for populating the stream.
 117          * @sa puts
 118          */
 119         virtual int loadFile(const char*);
 120 
 121         /*
 122          * @brief Write stream out to a file
 123          * @returns 1 on error, 0 on success
 124          * @param fd File descriptor to write to
 125          * @note The fd is not closed.
 126          */
 127         virtual int writeFile(const int) const;
 128 
 129         /*
 130          * @brief Write stream out to a file
 131          * @returns 1 on error, 0 on success
 132          * @param fname Filename to write to
 133          * @param mode Optional param to specify mode for new file
 134          */
 135         virtual int writeFile(const char*, mode_t mode = (S_IRUSR|S_IWUSR)) const;
 136 
 137 
 138         inline operator String() const { return str; };
 139         inline size_t length() const { return str.length(); };
 140         inline size_t capacity() const { return str.capacity(); };
 141         inline bool operator ! () const { return str.isEmpty(); };
 142 
 143         friend Stream& operator << (Stream&, const String&);
 144 };
 145 
 146 inline Stream& operator << (Stream& stream, const String& string) { stream.puts(string); return stream; }
