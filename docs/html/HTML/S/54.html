  42 /**
  43  * @brief Compare our String object with another String object, but only n characters
  44  * @param str The String object to compare to.
  45  * @param n The number of characters to compare.
  46  * @param start The index to start looking from
  47  * @return an integer less than, equal to, or greater than zero if our buffer is found, respectively, to be less than, to match, or be greater than str.
  48 */
  49 int String::compare(const String &str, size_t n, size_t start) const
  50 {
  51   const size_t my_len = length();
  52   /* Same string? */
  53   if (data() == str.data() && my_len == str.length())
  54     return 0;
  55 
  56   const size_t slen = n ? std::min(str.length(), n) : str.length();
  57   const size_t len = std::min(my_len - start, slen);
  58   const int diff = memcmp(begin() + start, str.begin(), len);
  59   if (diff)
  60     return diff;
  61   else if (n)
  62     return std::min(my_len - start, n) - slen;
  63   else
  64     return (my_len - start) - slen;
  65 }

  69 /**
  70  * @brief Insert a cstring at the given index.
  71  * @param string The cstring to be inserted.
  72  * @param pos The index to insert at.
  73  * @param n Up to this many characters will be used from the string.
  74  * @post A buffer is allocated.
  75  * @post If the old buffer was too small, it is enlarged.
  76  * @post The string is inserted at the given index.
  77  */
  78 void String::insert(size_t pos, const char *string, size_t n)
  79 {
  80   if (n == 0) return;
  81   if (pos && !hasIndex(pos-1)) return;
  82   
  83   size_t slen = (n == npos) ? strlen(string) : n;
  84 
  85   AboutToModify(length() + slen);
  86   memmove(Buf() + pos + slen, Buf() + pos, length() - pos);
  87   std::copy(string, string + slen, Buf() + pos);
  88   addLength(slen);
  89 }

  91 /**
  92  * @brief Replaces the given index with the given cstring up to the given amount of characters.
  93  * @param pos The index to replace at.
  94  * @param string The cstring to replace with.
  95  * @param n How many characters to use from string.
  96  */
  97 void String::replace(size_t pos, const char *string, size_t n)
  98 {
  99   if (n == 0) return;
 100   if (pos && !hasIndex(pos-1)) return;
 101 
 102   size_t slen = (n == npos) ? strlen(string) : n;
 103   size_t newlen = pos + slen;
 104   
 105   if (newlen >= length()) {
 106     AboutToModify(newlen);
 107   } else {
 108     newlen = length();
 109     getOwnCopy();
 110   }
 111   std::copy(string, string + slen, Buf() + pos);
 112   setLength(newlen);
 113 }

 115 std::istream& operator >> (std::istream& is, String &string) {
 116   char ch;
 117   string = "";    // empty string, will build one char at-a-time
 118   is >> ch;    // whitespace skipped, first non-white char in ch
 119 
 120   if (!is.fail())
 121   {
 122     do
 123     {
 124       string += ch;
 125       is.get(ch);
 126     } while (!is.fail() && !isspace(ch));
 127 
 128     if (isspace(ch))    // put whitespace back on the stream
 129     {
 130       is.putback(ch);
 131     }
 132   }
 133   return is;
 134 }

 147 /*
 148  * @brief Split a string by a delimiter
 149  * @param delim The delimiter to use
 150  * @param limit Optional param which specified the max number of elements to return
 151  * @note Excess whitespace is trimmed similar to Ruby's split.
 152  */
 153 Array<String> String::split(const String& delim, size_t limit) const {
 154   if (!length()) return Array<String>();
 155   if (limit == 1) {
 156     String list[] = {*this};
 157     return Array<String>(list, 1);
 158   }
 159 
 160   const String space(' ');
 161   Array<String> array;
 162   // Use a temporary - is a fast reference and never modified, so no COW ever used.
 163   String str(*this);
 164   size_t pos = 0;
 165 
 166   while (array.size() < limit - 1) {
 167     // Trim out left whitespace
 168     if (delim == space) while (str.length() && str[0] == ' ') ++str;
 169 
 170     // All done when there's nothing left
 171     if (!str) break;
 172 
 173     if ((pos = str.find(delim)) == npos)
 174       pos = str.length();
 175     array << str(0, pos);
 176     str += pos + delim.length();
 177   }
 178 
 179   // Add on extra
 180   if (limit != npos) {
 181     // Trim out left whitespace
 182     if (delim == space) while (str.length() && str[0] == ' ') ++str;
 183     array << str;
 184   }
 185 
 186   return array;
 187 }

 216 String& String::chomp() {
 217   if (length() && (*this)[length() - 1] == '\n') --(*this);
 218   if (length() && (*this)[length() - 1] == '\r') --(*this);
 219   return *this;

 230 size_t String::find (const String& str) const {
 231   if (length() >= str.length()) {
 232     const size_t last_pos = length() - str.length();
 233     for (size_t pos = 0; pos <= last_pos; ++pos)
 234       if (str[0] == (*this)[pos] && !memcmp(begin() + pos, str.begin(), std::min(str.length(), length() - pos)))
 235         return pos;
 236   }
 237   return npos;
 238 }

 240 size_t String::rfind (const String& str, const size_t lpos) const {
 241   if (length() >= str.length()) {
 242     const size_t last_pos = length() - str.length();
 243     for (size_t pos = last_pos; pos + 1 > lpos; --pos)
 244       if (str[0] == (*this)[pos] && !memcmp(begin() + pos, str.begin(), std::min(str.length(), length() - pos)))
 245         return pos;
 246   }
 247   return npos;
 248 }

 250 String String::sub(const String& search, const String& replacement, int limit) const {
 251   String newStr(length()), search_str(*this);
 252   size_t pos = 0;
 253   int cnt = 0;
 254   // Search in our const string so that stuff like \->\\ doesnt become \\\, \\\\ etc.
 255   while ((pos = search_str.find(search)) != npos) {
 256     newStr += search_str(0, pos);
 257     newStr += replacement;
 258     search_str += pos + search.length();
 259     if (limit != -1 && ++cnt == limit) break;
 260   }
 261   // Left over
 262   if (search_str.length())
 263     newStr += search_str;
 264   return newStr;
 265 }

 268   String newStr(*this);

 270   Array<String> values(hashes.values());
 271   for (size_t i = 0; i < keys.length(); ++i) {
 272     const String key(keys[i]), value(values[i]);
 273     //::printf("%s -> %s\n", key.c_str(), value.c_str());

 279 /**
 280  * @brief Splits off tokens
 281  * @param str The string to split a token off of
 282  * @param delim The delimiter to tokenize with
 283  * This will modify the given string to remove the first token, which is returned.
 284  * If no more tokens exist, the remaining string is returned and the string becomes "". Future calls will return a blank String("")
 285  */
 286 String newsplit(String& str, char delim)
 287 {
 288   if (!str.length()) return "";
 289   size_t pos = str.find(delim);
 290   if (pos == String::npos)
 291     pos = str.length();
 292 
 293   String first(str(0, pos));
 294 
 295   /* Trim out runs of whitespaces */
 296   if (delim == ' ') {
 297     while ((str.begin() + pos + 1) < str.end() && str[pos + 1] == ' ')
 298       ++pos;
 299   }
 300 
 301 //  str = str(pos + 1, str.length() - pos + 1);
 302   str += pos + 1;
 303   return first;
 304 }
