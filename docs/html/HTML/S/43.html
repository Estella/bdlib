  35 template <class T>
  36 /**
  37  * @class ArrayRef
  38  * @brief Helps the String class with reference counting
  39  * @todo look into something like a string object made of string pointers, and << displays all of the pointed to objects. (this would save copying to construct immuatable concatenated strings, but would only work with the << stream
  40  */
  41 class ArrayRef {
  42   public:
  43     ~ArrayRef() { FreeBuf(buf); };
  44     mutable size_t size; //Capacity of buffer
  45     mutable T* buf;
  46     mutable int n; //References
  47 
  48     ArrayRef() : size(0), buf(NULL), n(1) {};
  49     /**
  50      * @brief Ensure that the buffer capacity() is >= newSize; else grow/copy into larger buffer.
  51      * @param newSize A size that we need to Allocate the buffer to.
  52      * @param offset The offset of the old buffer so we know where to start
  53      * @pre newSize is > 0 (assumed as size_t is unsigned)
  54      * @post The buffer is at least nsize bytes long.
  55      * @post If the buffer had to grow, the old data was deep copied into the new buffer and the old deleted.
  56      */
  57     void Reserve(const size_t newSize, size_t& offset, size_t sublen) const
  58     {
  59       /* Don't new if we already have enough room! */
  60       if (size < newSize) {
  61         size = std::max(size_t(size * 1.5), newSize);
  62 
  63         T *newbuf = AllocBuf(size);
  64 
  65         if (newbuf != buf) {
  66           /* Copy old buffer into new - only copy the substring */
  67           std::copy(buf + offset, buf + offset + sublen, newbuf);
  68           FreeBuf(buf);
  69           buf = newbuf;
  70           offset = 0;
  71         }
  72       } else if ((size - offset) < newSize) {
  73         // There's enough room in the current buffer, but we're offsetted/shifted to a point where there's no room left
  74         // Shift everything to the beginning and reset the offset
  75         /* Only copy the substring */
  76         memmove(buf, buf + offset, sublen);
  77         offset = 0;
  78       }
  79     }
  80 
  81     /**
  82      * @brief Allocates a buffer and returns it's address.
  83      * @param bytes The number of bytes to allocate.
  84      * @post A new block of memory is allocated.
  85      * @todo Implement mempool here.
  86      */
  87     inline T* AllocBuf(const size_t bytes) const {
  88       return new T[bytes];
  89     }
  90 
  91     /**
  92      * @brief Free's up the allocated buffer.
  93      * @param p The buffer to be free'd
  94      * @post The buffer is deleted.
  95      * @todo Implement mempool here.
  96      */
  97     inline void FreeBuf(T* p) const {
  98       delete[] p;
  99     }
 100 
 101     /**
 102      * @brief Is this string shared?
 103      */
 104     inline bool isShared() const { return n > 1; };
 105   private:
 106     // No copying allowed
 107     ArrayRef(const ArrayRef&); ///<Block implicit copy constructor
 108     ArrayRef& operator=(const ArrayRef&); ///<Block implicit copy constructor
 109 };

 164 class ReferenceCountedArray {
 165   public:
 166     typedef T value_type;
 167 
 168     typedef size_t             size_type;
 169     typedef ptrdiff_t          difference_type;
 170     typedef value_type*        pointer;
 171     typedef const value_type*  const_pointer;
 172     typedef value_type&        reference;
 173     typedef const value_type&  const_reference;
 174 
 175     static const size_t npos = static_cast<size_t>(-1);
 176 
 177   private:
 178     /**
 179      * @brief Detach from the shared reference.
 180      * This is only called when losing the old buffer or when modifying the buffer (and copy-on-write is used)
 181      * @note This does not free the old reference, as it is still in use
 182      */
 183     void doDetach() const {
 184       decRef();
 185       Ref = new ArrayRef<value_type>();
 186       sublen = 0;
 187       offset = 0;
 188       my_hash = 0;
 189     }
 190     /**
 191      * @brief Increment our reference counter.
 192      */
 193     inline int incRef() const { return ++Ref->n; };
 194 
 195     /**
 196      * @brief Decrement our reference counter.
 197      */
 198     inline int decRef() const { return --Ref->n; };
 199 
 200   protected:
 201     /**
 202      * @brief Set the lengths to the specified length
 203      * @param newLen the new length to set to
 204      */
 205     inline void setLength(size_t newLen) const { sublen = newLen; my_hash = 0; };
 206 
 207     /**
 208      * @sa setLength()
 209      */
 210     inline void addLength(size_t diff) const { sublen += diff; my_hash = 0; };
 211 
 212     /**
 213      * @sa setLength()
 214      */
 215     inline void subLength(size_t diff) const { sublen -= diff; my_hash = 0; };
 216 
 217     /**
 218      * @brief Mutable Ref->buf+offset reference for use internally
 219      */
 220     //pointer mdata() const { return Buf() + offset; };

 222     /**
 223      * @brief Mutable Ref->buf reference for use internally
 224      */
 225     inline pointer Buf(size_t pos = 0) const { return &Ref->buf[offset + pos]; };
 226 
 227     /**
 228      * @brief Ref->buf reference for use internally
 229      */
 230     inline const_pointer constBuf(size_t pos = 0) const { return Buf(pos); };
 231   private:
 232     /**
 233      * @brief The array reference for reference counting
 234      * This is mutable so that Ref->n can be modified, which really is mutable
 235      */
 236     mutable ArrayRef<value_type> *Ref;
 237   protected:
 238     /**
 239      * Return the real buffer's start point, without accounting for offset. This is used for cleaning the buffer when needed.
 240      */
 241     inline const_pointer real_begin() const { return Ref->buf; };
 242 
 243     /**
 244      * This is for subarrays: so we know where the subarray starts.
 245      */
 246     mutable size_t offset;
 247     /**
 248      * This is for subarrays: so we know where the subarray ends.
 249      */
 250     mutable size_t sublen;
 251 
 252     /**
 253      * Cache of current hash() result. 0 if stale
 254      */
 255     mutable size_t my_hash;
 256 
 257   private:
 258     /**
 259      * @brief Detach from the reference
 260      * This is called when the old buffer is no longer needed for this Array.
 261      * ie, operator=() was called.
 262      */
 263     void Detach() {
 264       if (isShared()) {
 265         doDetach();
 266       } else {
 267         setLength(0);
 268         offset = 0;
 269         my_hash = 0;
 270       }
 271     }
 272 
 273     /**
 274      * @brief Free up our reference if we have the last one.
 275      * @post The reference counter is decremented.
 276      * @post If this was the last Reference, it is free'd
 277      * This is only called in ~Array() and operator=(Array&).
 278      * It checks whether of not this Array was the last reference to the buffer, and if it was, it removes it.
 279      */
 280     inline void CheckDeallocRef() {
 281       if (decRef() < 1)
 282         delete Ref;
 283     }
 284 
 285     //    void COW(size_t) const;
 286     /**
 287      * @brief Ensure that our internal buffer is unshared.
 288      * @param n Create/Grow the buffer to this size.
 289      * @pre n is > 0; this is assumed due to size_t being unsigned though.
 290      * @post The internal buffer/data is unshared
 291      * @post The buffer is at least size n.
 292      * @post The buffer is deep copied to a new buffer.
 293      *
 294      * Ensure that our internal buffer is unshared.
 295      * If needed, performs a deep copy into a new buffer (COW).
 296      * Also take a hint size n of the new string's size as to avoid needless copying/allocing.
 297      */
 298     void COW(size_t n) const {
 299       const_pointer oldBuf = constBuf();
 300       size_t oldLength = length();
 301 
 302       doDetach(); //Detach from the shared reference
 303       Reserve( std::max(oldLength, n) ); //Will set capacity()/size
 304       std::copy(oldBuf, oldBuf + oldLength, Buf());
 305       setLength(oldLength);
 306     }
 307 
 308   protected:
 309     /**
 310      * @brief Force COW if needed
 311      * @post The array is no longer shared, if it was.
 312      */
 313     inline void getOwnCopy() const { AboutToModify(length()); };
 314 
 315     /**
 316      * @brief Warn the reference counting that it may need to COW
 317      * @post The buffer is detached/COW, and possibly larger
 318      * @todo If the buffer is shared and needs to shrink, the sublen should just be decreased.
 319      */
 320     inline void AboutToModify(size_t n) const {
 321       my_hash = 0;
 322       if (isShared())
 323         COW(n); // Clears the offset
 324       else {
 325         Reserve(n);
 326         /* Shift the offset away */
 327       }
 328     }
 329   public:
 330     ReferenceCountedArray() : Ref(new ArrayRef<value_type>()), offset(0), sublen(0), my_hash(0) {};
 331     ReferenceCountedArray(const ReferenceCountedArray& rca) : Ref(rca.Ref), offset(rca.offset), sublen(rca.sublen), my_hash(rca.my_hash) { incRef(); };
 332     /**
 333      * @brief Create an empty container with at least the specified bytes in size.
 334      * @param newSize Reserve at least this many bytes for this String.
 335      * @post This string's memory will also never be shrunk.
 336      * @post A buffer has been created.
 337      *
 338      * The idea behind this is that if a specific size was asked for, the buffer is like
 339      * a char buf[N];
 340      */
 341     explicit ReferenceCountedArray(const size_t newSize) : Ref(new ArrayRef<value_type>()), offset(0), sublen(0), my_hash(0) {
 342       if (newSize <= 0) return;
 343       Reserve(newSize);
 344     };
 345     /**
 346      * @brief Create a container filled with n copies of the given value.
 347      * @param newSize Reserve at least this many bytes for this String.
 348      * @param value The value to populate the array with
 349      * @post This string's memory will also never be shrunk.
 350      * @post A buffer has been created.
 351      *
 352      */
 353     ReferenceCountedArray(const size_t newSize, const value_type value) : Ref(new ArrayRef<value_type>()), offset(0), sublen(0), my_hash(0) {
 354       if (newSize <= 0) return;
 355       Reserve(newSize);
 356 
 357       for (size_t i = 0; i < newSize; ++i)
 358         *(Buf(i)) = value;
 359       this->setLength(newSize);
 360     }

 363      * @brief Array Destructor
 364      * @post If the Array's Reference is not shared, it is free'd.
 365      * @post If the Array's Reference IS shared, it is decremented and detached.

 367     virtual ~ReferenceCountedArray() { CheckDeallocRef(); };

 369     /**
 370      * @brief Sets our Reference to the given String reference.
 371      * @param rca The String object to reference.
 372      * @post The old buffer (if we had one) is free'd.
 373      * @post Our Reference now points to the given String.
 374      * @post Our old rca object has been deleted (disconnected).
 375      * @return The new rca object.
 376      * This handles self-assignment just fine, checking for it explicitly would be ineffecient for most cases.
 377      */
 378     const ReferenceCountedArray& operator=(const ReferenceCountedArray &rca) {
 379       rca.incRef();
 380       offset = rca.offset;
 381       sublen = rca.sublen;
 382       my_hash = rca.my_hash;
 383       CheckDeallocRef();

 428     /**
 429      * @brief Resize the array to the given length.
 430      * @param len The length to resize to.
 431      * @param value The optional parameter to fill the space with if the array is expanded
 432      */
 433     void resize(size_t len, value_type value = value_type()) {
 434       if (len < this->length()) {
 435         this->subLength(this->length() - len);
 436       } else {
 437         this->AboutToModify(len);
 438         for (size_t i = 0; i < (len - this->length()); ++i)
 439           *(Buf(this->length() + i)) = value;
 440         this->addLength(len - this->length());
 441       }
 442     }

 475     typedef Hash<value_type> HashType;

 477    /**
 478      * @brief Return a hash of every element in the array. Cache result as well.
 479      * @note DJB's hash function
 480      */
 481     virtual size_t hash() const {
 482       if (my_hash != 0) return my_hash;
 483       HashType hasher;
 484       size_t _hash = 5381;
 485 
 486       for(size_t i = 0; i < this->length(); ++i)
 487         _hash = ((_hash << 5) + _hash) + hasher(this->data()[i]);
 488       return (my_hash = (_hash & 0x7FFFFFFF));
 489     }

 610     /**
 611      * @brief Return a new array from a subarray
 612      * @return a new ReferenceCountedArray
 613      * @param start The offset to begin the subarray from (indexed from 0)
 614      * @param len The length of the subarray to return
 615      * The returned slice is a reference to the original array until modified.
 616      */
 617     void slice(int start, int len = -1) {
 618       if (len == -1) len = int(length()) - start;
 619       // Start is after the end, set us to an empty array
 620       if (start >= static_cast<signed>(length())) {
 621         offset = length();
 622         setLength(0);
 623         return;
 624       }
 625 
 626       //Count backwards from the end
 627       if (start < 0)
 628         start = length() + start;
 629       // Start was before the beginning, just reset to the beginning
 630       if (start < 0)
 631         start = 0;
 632 
 633       offset += start;
 634       //If the length of the subarray exceeds the end of the array, truncate to the end of the array
 635       if (start + len >= static_cast<signed>(length()))
 636         len = length() - start;
 637       // If the length is negative, stop from counting backwards from the end
 638       else if (len < 0)
 639         len = length() - start + len;
 640       setLength(len);
 641     }

 661     /**
 662      * @brief Inserts a ReferenceCountedArray object into our buffer
 663      * @param pos The index to insert at.
 664      * @param rca The rca to insert.
 665      * @param n The length to insert.
 666      * @post The buffer contains n items from rca inserted at index pos.
 667      */
 668     void insert(size_t pos, const ReferenceCountedArray& rca, size_t n = npos) {
 669       if (n == 0) return;
 670       if (pos && !hasIndex(pos-1)) return;
 671 
 672       size_t slen = rca.length();
 673 
 674       /* New rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 675       if (pos == 0 && slen > length() && (n == npos || n == slen)) {
 676         *this = rca;
 677         return;
 678       }
 679 
 680       if (n == npos || n > slen)
 681         n = slen;
 682       slen -= slen - n;
 683       AboutToModify(length() + slen);
 684       memmove(Buf() + pos + slen, Buf() + pos, length() - pos);
 685       std::copy(rca.begin(), rca.begin() + slen, Buf() + pos);
 686       addLength(slen);
 687     }
 688 
 689     /**
 690      * @brief Insert an item at the given index.
 691      * @param pos The index to insert at.
 692      * @param item The item to be inserted.
 693      * @post A buffer is allocated.
 694      * @post If the old buffer was too small, it is enlarged.
 695      * @post The item is inserted at the given index.
 696      */
 697     void insert(size_t pos, const_reference item)
 698     {
 699       if (pos && !hasIndex(pos-1)) return;
 700 
 701       AboutToModify(length() + 1);
 702       memmove(Buf() + pos + 1, Buf() + pos, length() - pos);
 703       *(Buf(pos)) = item;
 704       addLength(1);
 705     }
 706 
 707     /**
 708      * @brief Replace the given index with the given item.
 709      * @param pos The index to replace.
 710      * @param item The item to replace with.
 711      * @post The given index has been replaced.
 712      * @post COW is done if needed.
 713      */
 714     void replace(size_t pos, const_reference item) {
 715       if (pos && !hasIndex(pos-1)) return;
 716 
 717       getOwnCopy();
 718       *(Buf(pos)) = item;
 719     }
 720 
 721     /**
 722      * @brief Replaces n elements in our buffer at index pos with the given ReferenceCountedArray object
 723      * @param pos The index to replace at.
 724      * @param rca The ReferenceCountedArray object to replace with.
 725      * @param n The number of characters to use for the replace.
 726      */
 727     void replace(size_t pos, const ReferenceCountedArray &rca, size_t n = npos) {
 728       if (n == 0) return;
 729       if (pos && !hasIndex(pos-1)) return;
 730 
 731       size_t slen = rca.length();
 732 
 733       /* Replace rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 734       if (pos == 0 && slen > length() && (n == npos || n == slen)) {
 735         *this = rca;
 736         return;
 737       }
 738 
 739       if (n == npos || n > slen)
 740         n = slen;
 741       slen -= slen - n;
 742 
 743       size_t newlen = pos + slen;
 744 
 745       if (newlen >= length()) {
 746         AboutToModify(newlen);
 747       } else {
 748         newlen = length();
 749         getOwnCopy();
 750       }
 751       std::copy(rca.begin(), rca.begin() + slen, Buf() + pos);
 752       setLength(newlen);
 753     }
