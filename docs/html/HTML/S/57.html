  39 template <class T>
  40 /**
  41  * @class Array
  42  * @brief Dynamic Array
  43  */
  44 class Array : public ReferenceCountedArray<T> {
  45   public:
  46 //    friend class ArrayIterator<T>;
  47 //    friend class ArrayConstIterator<T>;
  48 
  49 //    typedef ArrayIterator<T> iterator;
  50 //    typedef ArrayConstIterator<T> const_iterator;
  51     typedef T value_type;
  52 
  53     typedef size_t             size_type;
  54     typedef ptrdiff_t          difference_type;
  55     typedef value_type*        pointer;
  56     typedef const value_type*  const_pointer;
  57     typedef value_type&        reference;
  58     typedef const value_type&  const_reference;
  59 
  60     /* Constructors */
  61     Array() : ReferenceCountedArray<value_type>() {};
  62     Array(const Array<value_type>& array) : ReferenceCountedArray<value_type>(array) {};
  63     /**
  64      * @brief Create a Array from a given carray.
  65      * @param carray The null-terminated array to create the object from.
  66      * @post A ArrayBuf has been initialized.
  67      * @post The buffer has been filled with the array.
  68      * @test Array test("Some array");
  69      */
  70     Array(const_pointer carray) : ReferenceCountedArray<value_type>() {
  71       size_t i = 0;
  72       while (1) {
  73         if (!carray[i]) break;
  74         push(carray[i]);
  75         ++i;
  76       }
  77     };
  78 
  79     /**
  80      * @brief Create a Array from a given carray.
  81      * @param carray The null-terminated array to create the object from.
  82      * @param len How big is the carray?
  83      * @post A ArrayBuf has been initialized.
  84      * @post The buffer has been filled with the array.
  85      * @test Array test("Some array");
  86      */
  87     Array(const_pointer carray, size_t len) : ReferenceCountedArray<value_type>() {
  88       this->Reserve(len);
  89       for (size_t i = 0; i < len; ++i)
  90         push(carray[i]);
  91     };
  92 
  93     /**
  94      * @brief Create an empty Array container with at least the specified elements in size.
  95      * @param newSize Reserve at least this many buckets for this Array.
  96      * @post This array's memory will also never be shrunk.
  97      * @post A buffer has been created.
  98      */
  99     explicit Array(const size_type newSize) : ReferenceCountedArray<value_type>(newSize) {};
 100 
 101     virtual ~Array() {};
 102 
 103     /*
 104      * @brief Add an item to the end of the array
 105      */
 106     inline void push(const value_type item) {
 107       AboutToModify(this->length() + 1);
 108       *(Buf(this->length())) = item;
 109       this->addLength(1);
 110     }
 111 
 112     /**
 113      * @sa push
 114      */
 115     inline friend Array<value_type>& operator<< (Array<value_type>& array, const_reference item) {
 116       array.push(item);
 117       return array;
 118     }
 119 
 120     /*
 121      * @brief Shift a value off the end of the array
 122      * @return the item
 123      */
 124     value_type pop() {
 125       if (this->isEmpty()) return value_type();
 126 
 127       value_type temp(*(Buf(this->length() - 1)));
 128       this->subLength(1);
 129       return temp;
 130     }
 131 
 132     /**
 133      * @sa pop
 134      */
 135     inline friend Array<value_type>& operator>> (Array<value_type>& array, reference item) {
 136       item = array.pop();
 137       return array;
 138     }
 139 
 140     /*
 141      * @brief Join an array by a delimiter into a string
 142      */
 143     String join(const String& delim) const {
 144       if (!this->length()) return String();
 145       String str(*(this->Buf(0)));
 146       for (size_t i = 1; i < this->length(); ++i) {
 147         str += delim + *(this->Buf(i));
 148       }
 149       return str;
 150     }
 151 
 152     inline bool equals(const Array& array) const { return equals(array, array.length()); };
 153     /**
 154      * @brief Compare our Array object with another Array object, but only n elements
 155      * @param array The Array object to equals to.
 156      * @param n The number of items to equals.
 157      * @return True if the number of elements are the same, and they all are equal.
 158      */
 159     bool equals(const Array &array, size_t n) const
 160     {
 161       size_t my_len = this->length();
 162       bool same_length = (my_len == array.length());
 163 
 164       /* Same array? */
 165       if (this->data() == array.data() && same_length)
 166         return true;
 167 
 168       if (!same_length)
 169         return false;
 170 
 171       size_t slen = std::min(array.length(), n);
 172       size_t len = std::min(my_len, slen);
 173 
 174       for (size_t i = 0; i < len; ++i) {
 175         if ((*this)[i] != array[i])
 176           return false;
 177       }
 178       return true;
 179     }
 180 
 181     inline friend bool operator == (const Array& lhs, const Array& rhs) { return lhs.equals(rhs); };
 182     inline friend bool operator != (const Array& lhs, const Array& rhs) {return ! (lhs == rhs);};
 183 
 184     // Subarrays
 185     /*
 186      * @sa ReferenceCountedArray::slice()
 187      */
 188     inline Array subarray(int start, int len) const {
 189       Array newArray(*this);
 190       newArray.slice(start, len);
 191       return newArray;
 192     };
 193 
 194     /*
 195      * @sa substring
 196      */
 197     inline Array operator()(int start, int len) const { return subarray(start, len); };
 198 
 199     /**
 200      * @brief Returns a 'Slice' class for safe (cow) writing into the array
 201      * @sa Slice
 202      * @param start Starting position
 203      * @param len How many items to use
 204      */
 205     inline Slice<Array> operator()(int start, int len) { return Slice<Array>(*this, start, len); };

 207 #ifdef CPPUNIT_VERSION
 208     void CPPUNIT_checkArrayEqual(Array actual, CPPUNIT_NS::SourceLine sourceLine) {
 209       if ((*this) == actual) return;
 210       ::CPPUNIT_NS::Asserter::failNotEqual(this->join("|").c_str(), actual.join("|").c_str(), sourceLine);
 211     }
 212 #endif /* CPPUNIT_VERSION */
 213 };
