  53     typedef size_t             size_type;
  54     typedef ptrdiff_t          difference_type;
  55     typedef value_type*        pointer;
  56     typedef const value_type*  const_pointer;
  57     typedef value_type&        reference;
  58     typedef const value_type&  const_reference;
  59 
  60     /* Constructors */
  61     Array() : ReferenceCountedArray<value_type>() {};
  62     Array(const Array<value_type>& array) : ReferenceCountedArray<value_type>(array) {};
  63     /**
  64      * @brief Create a Array from a given carray.
  65      * @param carray The null-terminated array to create the object from.
  66      * @post A ArrayBuf has been initialized.
  67      * @post The buffer has been filled with the array.
  68      * @test Array test("Some array");
  69      */
  70     Array(const_pointer carray) : ReferenceCountedArray<value_type>() {
  71       size_t i = 0;
  72       while (1) {
  73         if (!carray[i]) break;
  74         push(carray[i]);
  75         ++i;
  76       }
  77     };
  78 
  79     /**
  80      * @brief Create a Array from a given carray.
  81      * @param carray The null-terminated array to create the object from.
  82      * @post A ArrayBuf has been initialized.
  83      * @post The buffer has been filled with the array.
  84      * @test Array test("Some array");
  85      */
  86     Array(const_pointer carray, size_t len) : ReferenceCountedArray<value_type>() {
  87       this->Reserve(len);
  88       for (size_t i = 0; i < len; ++i)
  89         push(carray[i]);
  90     };
  91 
  92     /**
  93      * @brief Create an empty Array container with at least the specified elements in size.
  94      * @param newSize Reserve at least this many buckets for this Array.
  95      * @post This array's memory will also never be shrunk.
  96      * @post A buffer has been created.
  97      */
  98     explicit Array(const size_type newSize) : ReferenceCountedArray<value_type>(newSize) {};
  99 
 100     virtual ~Array() {};
 101 
 102     virtual size_t hash() const {return 0;};
 103 
 104     /*
 105      * @brief Add an item to the end of the array
 106      */
 107     inline void push(const value_type item) {
 108       AboutToModify(this->length() + 1);
 109       *(Buf(this->length())) = item;
 110       this->addLength(1);
 111     }
 112 
 113     /**
 114      * @sa push
 115      */
 116     inline friend Array<value_type>& operator<< (Array<value_type>& array, const_reference item) {
 117       array.push(item);
 118       return array;
 119     }
 120 
 121     /*
 122      * @brief Shift a value off the end of the array
 123      * @return the item
 124      */
 125     value_type pop() {
 126       if (this->isEmpty()) return value_type();
 127 
 128       value_type temp(*(Buf(this->length() - 1)));
 129       this->subLength(1);
 130       return temp;
 131     }
 132 
 133     /**
 134      * @sa pop
 135      */
 136     inline friend Array<value_type>& operator>> (Array<value_type>& array, reference item) {
 137       item = array.pop();
 138       return array;
 139     }
 140 
 141     /*
 142      * @brief Join an array by a delimiter into a string
 143      */
 144     String join(const String& delim) const {
 145       if (!this->length()) return String();
 146       String str(*(this->Buf(0)));
 147       for (size_t i = 1; i < this->length(); ++i) {
 148         str += delim + *(this->Buf(i));
 149       }
 150       return str;
 151     }
 152 
 153     inline bool equals(const Array& array) const { return equals(array, array.length()); };
 154     /**
 155      * @brief Compare our Array object with another Array object, but only n elements
 156      * @param array The Array object to equals to.
 157      * @param n The number of items to equals.
 158      * @return True if the number of elements are the same, and they all are equal.
 159      */
 160     bool equals(const Array &array, size_t n) const
 161     {
 162       size_t my_len = this->length();
 163       bool same_length = (my_len == array.length());
 164 
 165       /* Same array? */
 166       if (this->data() == array.data() && same_length)
 167         return true;
 168 
 169       if (!same_length)
 170         return false;
 171 
 172       size_t slen = std::min(array.length(), n);
 173       size_t len = std::min(my_len, slen);
 174 
 175       for (size_t i = 0; i < len; ++i) {
 176         if ((*this)[i] != array[i])
 177           return false;
 178       }
 179       return true;
 180     }
 181

 183     inline friend bool operator != (const Array& lhs, const Array& rhs) {return ! (lhs == rhs);};
