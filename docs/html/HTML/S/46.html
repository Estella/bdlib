  73          * @brief Create a String from a given cstring.
  74          * @param cstring The null-terminated character array to create the object from.

  79         String(const char* cstring) : ReferenceCountedArray<String_Array_Type>() { if (cstring) append(cstring); };

  81         /**
  82          * @brief Create a String from a given cstring with the given strlen.

 295 /**
 296  * @relates String
 297  * @brief Concatenates two string objects together.
 298  * @param string1 The LHS string.
 299  * @param string2 The RHS string.
 300  * @post A new string is allocated, reference copied and returned.
 301  * @return Returns a new string that can be reference copied by the lvalue.
 302  */
 303 inline String operator + (const String& string1, const String& string2) {
 304   String temp(string1);
 305   temp += string2;
 306   return temp;

 360 inline String& String::operator += (const int n) {
 361   if (!length())
 362     return *this;
 363   if (int(length()) - n < 0) {
 364     offset = length();
 365     setLength(0);
 366   } else {
 367     offset += n;
 368     subLength(n);
 369   }
 370   return *this;
 371 }
 372 
 373 inline String& String::operator -= (const int n) {
 374   if (!length())
 375     return *this;
 376   if (int(length()) - n < 0) {
 377     offset = length();
 378     setLength(0);
 379   } else
 380     subLength(n);
 381   return *this;
 382 }
