  53 /**
  54  * @class String
  55  * @brief Provides custom string class for easy and optimized string manipulation.

  68         /* Constructors */
  69         String() : ReferenceCountedArray<String_Array_Type>() {};
  70         String(const String& string) : ReferenceCountedArray<String_Array_Type>(string) {};
  71         /**
  72          * @brief Create a String from a given cstring.
  73          * @param cstring The null-terminated character array to create the object from.
  74          * @post The buffer has been filled with the string.
  75          * @test String test("Some string");
  76         */
  77         String(const char* cstring) : ReferenceCountedArray<String_Array_Type>() { if (cstring) append(cstring); };
  78 
  79         /**
  80          * @brief Create a String from a given cstring with the given strlen.
  81          * @param cstring The null-terminated character array to create the object from.
  82          * @param slen The length of the given string to use.
  83          * @pre len > 0
  84          * @post The buffer has been filled with the string (up to len characters).
  85          * @test String test("Some string");
  86          */
  87         String(const char* cstring, size_t slen) : ReferenceCountedArray<String_Array_Type>() { append(cstring, slen); };

  89         /**
  90          * @brief Create a String from a given character.
  91          * @param ch The character to create the string from.
  92          * @post A stringBuf has been initialized.
  93          * @post The buffer has been filled with the caracter.
  94          * @test String test('a');
  95          */
  96         String(const char ch) : ReferenceCountedArray<String_Array_Type>() { append(ch); };

  98         /**
  99          * @brief Create an empty String container with at least the specified bytes in size.

 120          * @brief Cstring accessor
 121          * @return A null-terminated character array (cstring).
 122          * @post The buffer size is (possibly) incremented by 1 for the '\\0' character.

 293 /**
 294  * @relates String
 295  * @brief Concatenates two string objects together.
 296  * @param string1 The LHS string.
 297  * @param string2 The RHS string.
 298  * @post A new string is allocated, reference copied and returned.
 299  * @return Returns a new string that can be reference copied by the lvalue.
 300  */
 301 inline String operator + (const String& string1, const String& string2) {
 302   String temp(string1);
 303   temp += string2;
 304   return temp;
 305 }

 358 inline String& String::operator += (const int n) {
 359   if (!length())
 360     return *this;
 361   if (int(length()) - n < 0) {
 362     offset = length();
 363     setLength(0);
 364   } else {
 365     offset += n;
 366     subLength(n);
 367   }
 368   return *this;
 369 }
 370 
 371 inline String& String::operator -= (const int n) {
 372   if (!length())
 373     return *this;
 374   if (int(length()) - n < 0) {
 375     offset = length();
 376     setLength(0);
 377   } else
 378     subLength(n);
 379   return *this;
 380 }
