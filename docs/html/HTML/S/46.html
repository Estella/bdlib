  50 /**
  51   * @class StringBuf
  52   * @brief Helps the String class with reference counting
  53   * @todo look into something like a string object made of string pointers, and << displays all of the pointed to objects. (this would save copying to construct immuatable concatenated strings, but would only work with the << stream

  56 class StringBuf {
  57   public:
  58         ~StringBuf() { FreeBuf(buf); };
  59         mutable size_t len; //Length of string
  60         mutable size_t size; //Capacity of buffer
  61         mutable char* buf;
  62         mutable uint8_t n; //References
  63         mutable char sbuf[16];
  64     
  65         StringBuf() : len(0), size(0), buf(NULL), n(1) {};
  66         void Reserve(const size_t) const;
  67         /**
  68          * @brief Allocates a buffer and returns it's address.
  69          * @param bytes The number of bytes to allocate.
  70          * @post A new block of memory is allocated.
  71          * @todo Implement mempool here.
  72          */
  73         inline char* AllocBuf(const size_t bytes) const {
  74           if (bytes <= sizeof(sbuf)) return sbuf;
  75           else return new char[bytes];
  76         }
  77 
  78         /**
  79          * @brief Free's up the allocated buffer.
  80          * @param p The buffer to be free'd
  81          * @post The buffer is deleted.
  82          * @todo Implement mempool here.
  83          */
  84         inline void FreeBuf(const char* p) const {
  85           if (p != sbuf) delete[] p;
  86         }
  87 
  88         /**
  89           * @brief Is this string shared?
  90           */
  91         inline bool isShared() const { return n > 1; };
  92 private:
  93         // No copying allowed
  94         StringBuf(const StringBuf&); ///<Block implicit copy constructor
  95         StringBuf& operator=(const StringBuf&); ///<Block implicit copy constructor
  96 };

  99  * @class String

 106         /**
 107          * @class Cref
 108          * @brief Safe element reading and writing.
 109          * @todo This should not provide copy constructors for Cref, they shouldn't be needed because of const char String::operator[]
 110          * This class should be optimized away and fully inlined such that:
 111          * String s;
 112          * s[0] = 'a';
 113          * Should be rewritten as:
 114          * s.write(0, 'a');
 115          */
 116         class Cref {
 117             friend class String;
 118             String& s;
 119             int k;
 120 
 121             /**
 122               * @brief Used by String::Cref operator[]
 123               */
 124             Cref(String& string, int i) : s(string), k(i) {};
 125             Cref(); //Not defined - never used
 126 
 127           public:            
 128             Cref(const Cref& cref) : s(cref.s), k(cref.k) {};
 129             inline Cref& operator=(const Cref& cref) {
 130               (*this) = (char) cref;
 131               return (*this);
 132             }
 133 
 134           public:
 135             /**
 136              * @sa char String::operator[]
 137              */
 138             inline operator char() const { return s.read(k); };
 139 
 140             /**
 141              * Stroustrup shows using this as void with no return value, but that breaks chaining a[n] = b[n] = 'b';
 142              */
 143             inline Cref& operator=(char c) { 
 144               s.write(k, c); 
 145               return (*this);
 146             };
 147         };

 150          * @class SubString
 151          * @brief Safe substring reading and writing.
 152          * @todo This should not provide copy constructors for Cref, they shouldn't be needed because of const char String::operator[]
 153          * This class should be optimized away and fully inlined such that:
 154          * String s("look over there");
 155          * s(0, 4) = "blah"';
 156          * Should be rewritten as:
 157          * s.replace(0, "blah", 4);
 158          */
 159          class SubString {

 171             inline SubString& operator= (const SubString& substring) {
 172               (*this) = (String) substring;
 173               return (*this);
 174             }

 176

 178              * @brief return a new (const) substring
 179              */

 192               return (*this);

 278          * It checks whether of not this String was the last reference to the buffer, and if it was, it removes it.

 292             doDetach();
 293           } else {
 294             setLength(0);
 295             offset = 0;
 296           }
