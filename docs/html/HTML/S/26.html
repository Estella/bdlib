  31 template <class T>
  32 /**
  33   * @class List
  34   * @brief List data structure
  35   */
  36 class List {
  37   class ListIterator;
  38 
  39   public:
  40     typedef ListIterator iterator;
  41     typedef T iterator_type;
  42 
  43 
  44   protected:
  45     struct Node {
  46       iterator_type ptr;
  47       Node* next;
  48       Node* prev;
  49       
  50       Node(const iterator_type& p) : ptr(p), next(NULL), prev(NULL) {};
  51     };
  52 
  53     void deleteNode(Node* node) {
  54       Node* next = node->next; //NULL if node = tail
  55       Node* prev = node->prev; //NULL if node = head
  56 
  57       delete node;
  58 
  59       if (next) next->prev = prev;
  60       if (prev) prev->next = next;
  61 
  62       if (node == head) head = next;
  63       if (node == tail) tail = prev;
  64 
  65       --my_size;
  66     }
  67 
  68     Node* head;
  69     Node* tail;
  70     size_t my_size;
  71 
  72   public:
  73     List() : head(NULL), tail(NULL), my_size(0) {};
  74     virtual ~List() { clear(); };
  75 
  76     List(const List& list) : head(NULL), tail(NULL), my_size(0) {
  77       for (Node* search = list.head; search; search = search->next)
  78         insert(search->ptr);
  79     }
  80 
  81     List& operator =(const List& list) {
  82       if (&list != this) {
  83         clear();
  84         for (Node* search = list.head; search; search = search->next)
  85           insert(search->ptr);
  86       }
  87       return *this;
  88     }
  89 
  90     void clear() {
  91       Node* node = head, *node_n = NULL;
  92       while (node) {
  93         node_n = node->next;
  94         delete node;
  95         node = node_n;
  96       }
  97       head = tail = NULL;
  98       my_size = 0;
  99     };
 100 
 101     inline size_t size() const { return my_size; };
 102     inline bool isEmpty() const { return size() == 0; };
 103     //inline operator bool() const { return !isEmpty(); };
 104 
 105     /**
 106      * @brief Insert into the list at the head
 107      * @param ptr The ptr to insert
 108      */
 109     void insert(const iterator_type& ptr) {
 110       Node* node = new Node(ptr);
 111       if (!head) {
 112         head = tail = node;
 113       } else {
 114         head->prev = node;
 115         node->next = head;
 116         head = node;
 117       }
 118       ++my_size;
 119     }
 120 
 121     /**
 122      * @sa insert
 123      */
 124     inline friend List<iterator_type>& operator<< (List<iterator_type>& list, const iterator_type& item) {
 125       list.insert(item);
 126       return list;
 127     }
 128 
 129     bool contains(const iterator_type& ptr) const {
 130       if (head) {
 131         for (Node* node = head; node; node = node->next) {
 132           if (node->ptr == ptr)
 133             return 1;
 134         }
 135       }
 136       return 0;
 137     }
 138 
 139    const iterator_type find(const iterator_type& ptr) const {
 140       const iterator_type empty;
 141       if (isEmpty()) return empty;
 142 
 143       for (Node* current = head; current; current = current->next) {
 144         if (current->ptr == ptr)
 145           return current->ptr;
 146       }
 147       return empty;
 148     }
 149 
 150     /**
 151      * @brief Return a reference to the found element
 152      */
 153     iterator_type& findRef(const iterator_type& ptr) {
 154       Node* current = head;
 155       for (; current; current = current->next) {
 156         if (current->ptr == ptr)
 157           break;
 158       }
 159       return current->ptr;
 160     }
 161 
 162     bool remove(const iterator_type& ptr) {
 163       if (head) {
 164       /* cases (removing X):
 165        * 1) list = (X)
 166        * 2) list = (X)->(Y)
 167        * 3) list = (W)->(X)
 168        * 4) list = (W)->(X)->(Y)
 169        */
 170         for (Node* node = head; node; node = node->next) {
 171           if (node->ptr == ptr) {
 172             deleteNode(node);
 173             return true;
 174           }
 175         }
 176       }
 177       return false;
 178     }
 179 
 180     private:
 181     class ListIterator : public Iterator<iterator_type> {
 182       friend class List;
 183       private:
 184         List *list;
 185         Node *current;
 186         bool end;
 187 
 188         ListIterator(List &_list, bool _end = 0) : Iterator<iterator_type>(),
 189                                                               list(&_list),
 190                                                               current(_end ? _list.head : _list.tail), /* tail is beginning */
 191                                                               end(_end) {
 192         };
 193       public:
 194         ListIterator(const iterator& iter) : Iterator<iterator_type>(),
 195                                              list(iter.list),
 196                                              current(iter.current),
 197                                              end(iter.end) {
 198         };
 199         ListIterator() : Iterator<iterator_type>(),
 200                          list(NULL),
 201                          current(NULL),
 202                          end(0) {
 203         };
 204 
 205         iterator& operator =(const iterator& iter) {
 206           list = iter.list;
 207           current = iter.current;
 208           end = iter.end;
 209           return *this;
 210         }
 211 
 212         virtual ~ListIterator() { }
 213 
 214         virtual void remove() {
 215           Node *nextValue = end == 0 ? current->next : current->prev;
 216 
 217           list->deleteNode(current);
 218 
 219           current = nextValue;
 220         }
 221 
 222         virtual operator bool() { return (current != NULL); };
 223         virtual operator iterator_type () { return operator*(); };
 224           
 225         virtual iterator_type& operator *() { return (iterator_type&)current->ptr; }
 226 
 227         //Postfix
 228         virtual iterator operator ++(int) {
 229           iterator tmp(*this);
 230           current = current->prev;
 231           return tmp;
 232         }
 233 
 234         virtual iterator& operator ++() {
 235           current = current->prev;
 236           return *this;
 237         }
 238 
 239         //Postfix
 240         virtual iterator operator --(int) {
 241           iterator tmp(*this);
 242           current = current->next;
 243           return tmp;
 244         }
 245 
 246         virtual iterator& operator --() {
 247           current = current->next;
 248           return *this;
 249         }
 250     };
 251   public:
 252     inline iterator begin() { return iterator(*this); };
 253     inline iterator end() { return iterator(*this, 1); };
 254 };
