  27 #include "bdlib.h"

  30 BDLIB_NS_BEGIN
  31 
  32 template <class T>
  33 /**
  34  * @class Stack
  35  * @brief A FILO Stack, what else!
  36  */
  37 class Stack : public List<T> {
  38   typedef T iterator_type;
  39   typedef typename List<iterator_type>::Node Node;
  40 
  41   public:
  42     Stack() : List<iterator_type>() {};
  43     virtual ~Stack() {};
  44 
  45     /**
  46      * @brief Pop an element off the top of the stack
  47      */
  48     iterator_type pop() {
  49       iterator_type item = this->head->ptr;
  50       this->remove(item);
  51       return item;
  52     };
  53 
  54     /**
  55      * @sa pop
  56      */
  57     inline friend Stack<iterator_type>& operator>> (Stack<iterator_type>& stack, iterator_type& item) {
  58       item = stack.pop();
  59       return stack;
  60     }
  61 
  62     /**
  63      * @brief Peek at the first element, but don't pop it off
  64      * @todo Throw an exception?
  65      */
  66     inline const iterator_type peek() const {
  67       return this->isEmpty() ? iterator_type() : this->head->ptr;
  68     }
  69 
  70     /*
  71      * @brief Push an element on to the top of stack
  72      * @param item The element to push back
  73      */
  74     inline void push(const iterator_type &item) {
  75       this->insert(item);
  76     };
  77 
  78     /**
  79      * @sa push
  80      */
  81     inline friend Stack<iterator_type>& operator<< (Stack<iterator_type>& stack, const iterator_type& item) {
  82       stack.insert(item);
  83       return stack;
  84     }
  85 };
  86 
  87 BDLIB_NS_END
  88 #endif /* !_BD_STACK_H */
