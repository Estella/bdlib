  33 template <class Key, class Value>
  34 /**
  35   * @class BinaryTree
  36   * @brief Binary tree data structure
  37   */
  38 class BinaryTree {
  39   class BinaryTreeIterator;
  40 
  41   public:
  42     typedef BinaryTreeIterator iterator;
  43     typedef KeyValue<Key, Value> iterator_type;
  44 
  45   private:
  46     class Node {
  47       public:
  48         Node *left;
  49         Node *right;
  50         iterator_type kv;
  51 
  52         Node(const Key k, const Value v) : left(NULL), right(NULL), kv(k, v) { };
  53         Node(const Node& n) : left(n.left ? new Node(*(n.left)) : NULL),
  54                               right(n.right ? new Node(*(n.right)) : NULL),
  55                               kv(n.kv) {};
  56     };
  57 
  58     size_t my_size;
  59     Node *root;
  60 
  61     /** 
  62       * @brief Returns a reference to the node with a matching key
  63       * @param search Where to start the search in the tree
  64       * @param key The key to search for
  65       * @return The node searched for, if found, otherwise NULL
  66       */
  67     Node* &fetchNode(Node* const* search, const Key &key) const {
  68       while ((*search) != NULL) {
  69         if (key < (*search)->kv.key())
  70           search = &(*search)->left;
  71         else if (key > (*search)->kv.key())
  72           search = &(*search)->right;
  73         else {
  74           return (Node*&) *search;
  75         }
  76       }
  77       return (Node*&) *search;
  78     }
  79 
  80     /** 
  81       * @brief Insert a node into the given tree
  82       * @param search The tree to insert the node into
  83       * @param node The node to insert
  84       * @sa fetchNode
  85       * This first looks for the where the node SHOULD be in the tree.
  86       * It will of course fail, but this is the location to place the node.
  87       */
  88     inline void insertNode(Node** search, Node* const node) {
  89       Node* &insertAt = fetchNode(search, node->kv.key());
  90       insertNode(insertAt);
  91     }
  92 
  93     inline void insertNode(Node*& insertAt, Node* const node) {
  94       insertAt = node;
  95       ++my_size;
  96     }
  97 
  98     /**
  99       * @brief Delete a node from the tree. (possibly recursive)
 100       * @param node A reference to the node to delete.
 101       */
 102     void deleteNode(Node*& node) {
 103       if (node == NULL) 
 104         return;
 105       if (node->left == NULL) {
 106         Node* temp = node->right;
 107         delete node;
 108         node = temp;
 109         --my_size;
 110       } else if (node->right == NULL) {
 111         Node* temp = node->left;
 112         delete node;
 113         node = temp;
 114         --my_size;
 115       } else {
 116         //Two children, find max of left subtree and swap
 117         Node** temp = &(node->left);
 118 
 119         while ((*temp)->right != NULL)
 120           temp = &(*temp)->right;
 121         node->kv = (*temp)->kv;
 122         deleteNode(*temp);
 123       }
 124     }
 125   public:
 126     /** 
 127       * @brief Checks whether the given key is in the tree
 128       * @param key The key to search for
 129       * @return true/false depending on found/not-found
 130       */
 131     bool contains(const Key &key) const {
 132       if (isEmpty())
 133         return false;
 134       if (fetchNode(&root, key))
 135         return true;
 136       return false;
 137     }
 138 
 139     /**
 140       * @brief Returns the current size of the tree
 141       * @return The current size of the tree
 142       */
 143     inline size_t size() const { return my_size; };
 144     inline bool isEmpty() const { return size() == 0; };
 145     //inline operator bool() const { return !isEmpty(); };
 146 
 147   public:
 148     BinaryTree() : my_size(0), root(NULL) {};
 149     BinaryTree(const BinaryTree& tree) : my_size(tree.my_size), root(new Node(*(tree.root))) {};
 150 
 151     BinaryTree& operator = (const BinaryTree& tree) {
 152       if (&tree != this) {
 153         clear();
 154         my_size = tree.my_size;
 155         root = new Node(*(tree.root));
 156       }
 157       return *this;
 158     }
 159 
 160     /**
 161       * @brief Destructor
 162       * @sa deleteNode
 163       * This simply deletes the root node over and over until the root is NULL.
 164       * There may possibly be a quicker way that does not use recursion (deleteNode does) but this seems the simplest.
 165       */
 166     virtual ~BinaryTree() { clear(); };
 167 
 168     /**
 169       * @brief insert Key/Value pair into tree
 170       * @param key The key to be inserted (or replaced)
 171       * @param value The value to be inserted (or replaced)
 172       * @post The tree's size is increased by 1 if the element was not already in the tree
 173       */
 174     bool insert(const Key &key, const Value &value) {
 175       Node*& rNode = fetchNode(&root, key);
 176       if (rNode)
 177         rNode->kv = iterator_type(key, value);
 178       else {
 179         Node *newNode = new Node(key, value);
 180         insertNode(rNode, newNode);
 181       }
 182       return true;
 183     }
 184 
 185     /**
 186       * @brief Associate array type accessor (rvalue)
 187       * @sa getValue
 188       * @param key The key to search for
 189       */
 190     inline const Value operator [](const Key& key) const { return getValue(key); }
 191 
 192     /**
 193       * @brief Associate array type accessor (lvalue)
 194       * @param key The key to search for
 195       * @sa insert
 196       * @code tree["key"] = "value";
 197       * If the key is not in the tree, it is inserted, and the value set to the rvalue given.
 198       */
 199     Value& operator [](const Key& key) {
 200       Node*& node = fetchNode(&root, key);
 201       if (!node)
 202         insertNode(node, new Node(key, Value()));
 203       //The key was inserted at the return node's address! node is now guranteed to be NON-NULL ! */
 204       return node->kv.v;
 205     }
 206 
 207     /** 
 208       * @brief Remove the given key (and its value) from the tree
 209       * @param key The key to be searched/removed
 210       * @return Whether or not the key was removed
 211       */
 212     bool remove(const Key &key) {
 213       Node* &node = fetchNode(&root, key);
 214 
 215       if (node != NULL) {
 216         deleteNode(node);
 217         return true;
 218       }
 219       return false;
 220     }
 221 
 222     inline void clear() {
 223       while (root != NULL)
 224         deleteNode(root);
 225     }
 226 
 227     /**
 228       * @brief Find a value in the tree from the key given
 229       * @param key The key to search for
 230       * @return The value of the key searched for, or NULL if not found
 231       */
 232     Value getValue(const Key &key) const {
 233       const Value empty;
 234       if (isEmpty()) return empty;
 235 
 236       Node* node = fetchNode(&root, key);
 237       if (node)
 238         return node->kv.value();
 239       return empty;
 240     }

 248     Value getKey(const Value &value) const {
 249       const Key empty;

 252       for (iterator iter = begin(); iter; ++iter) {

 260   private:
 261     class BinaryTreeIterator : public Iterator<iterator_type> {
 262       friend class BinaryTree;
 263       typedef BinaryTree<Key, Value> Tree;
 264       private:
 265         Tree* tree;
 266         size_t index;
 267         size_t my_size;
 268         iterator_type* storage;
 269 
 270         void fillArray(int& i, const Node* node) {
 271           if (node == NULL) 
 272             return;
 273           fillArray(i, node->left);
 274           storage[i++] = iterator_type(node->kv);
 275           fillArray(i, node->right);
 276         }
 277 
 278         BinaryTreeIterator(Tree* t, Node* node, size_t size, bool end = 0) : Iterator<iterator_type>(), 
 279                                                                           tree(t),
 280                                                                           index(end ? size - 1: 0), 
 281                                                                           my_size(size), 
 282                                                                           storage(new iterator_type[size]) {
 283           int i = 0;
 284           fillArray(i, node);
 285         }
 286         //BinaryTreeIterator& operator=(const BinaryTreeIterator&); ///<Block implicit copy constructor
 287       public:
 288         BinaryTreeIterator(const iterator& iter) : Iterator<iterator_type>(), 
 289                                                    tree(iter.tree),
 290                                                    index(iter.index), 
 291                                                    my_size(iter.my_size), 
 292                                                    storage(new iterator_type[iter.my_size]) {
 293           for (size_t i = 0; i < my_size; ++i)
 294             storage[i] = iter.storage[i];
 295         }
 296 
 297         BinaryTreeIterator() : Iterator<iterator_type>(),
 298                                tree(NULL),
 299                                index(0),
 300                                my_size(0),
 301                                storage(NULL) {
 302         };
 303 
 304         iterator& operator =(const iterator& iter) {
 305           if (&iter != this) {
 306             if (my_size != iter.my_size) {
 307               delete[] storage;
 308               storage = new iterator_type[iter.my_size];
 309             }
 310 
 311             tree = iter.tree;
 312             index = iter.index;
 313             my_size = iter.my_size;
 314 
 315             for (size_t i = 0; i < my_size; ++i)
 316               storage[i] = iter.storage[i];
 317           }
 318           return *this;
 319         }

 321         virtual ~BinaryTreeIterator() { delete[] storage; }
 322

 324           tree->remove(storage[index].key());

 327           for (int i = index; i < my_size; ++i)
 328             storage[i] = storage[i + 1];
 329           //The iterator is 'invalid' now, but the index is shifter down in case iter++ is called
 330           --index;
 331 */
 332         }
 333 
 334         virtual operator bool() { return (index > 0 && index < my_size); };
 335 
 336         virtual operator iterator_type () { return operator*(); };
 337         virtual iterator_type& operator *() { return storage[index]; }
 338 
 339         //Postfix
 340         virtual iterator operator ++(int) {
 341           iterator tmp(*this);
 342           ++index;
 343           return tmp;
 344         }
 345 
 346         virtual iterator& operator ++() {
 347           ++index;
 348           return *this;
 349         }
 350 
 351         //Postfix
 352         virtual iterator operator --(int) {
 353           iterator tmp(*this);
 354           --index;
 355           return tmp;
 356         }
 357 
 358         virtual iterator& operator --() {
 359           --index;
 360           return *this;
 361         }
 362     };
 363   public:
 364     inline iterator begin() { return iterator(this, root, my_size); };
 365     inline iterator end() { return iterator(this, root, my_size, 1); };
